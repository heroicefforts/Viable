19  package net.heroicefforts.viable.android;
20  
21  import net.heroicefforts.viable.android.content.Issues;
22  import net.heroicefforts.viable.android.rep.IssueResource;
23  import net.heroicefforts.viable.android.rep.RepositoryFactory;
24  import android.content.Context;
25  import android.database.Cursor;
26  import android.graphics.drawable.Drawable;
27  import android.view.View;
28  import android.widget.ImageView;
29  import android.widget.SimpleCursorAdapter.ViewBinder;
30  
31  /**
32   * Binds issue values to the list view.
33   * 
34   * @author jevans
35   *
36   */
37  public class IssuesListViewBinder implements ViewBinder
38  {
39  	private Context ctx;
40  	private RepositoryFactory factory;
41  	
42  	public IssuesListViewBinder(Context ctx, RepositoryFactory factory)
43  	{
44  		this.ctx = ctx;
45  		this.factory = factory;
46  	}
47  	
48  	public boolean setViewValue(View view, Cursor cursor, int columnIndex)
49  	{
50  		String colName = cursor.getColumnName(columnIndex);
51  		if(Issues.ISSUE_PRIORITY.equals(colName) && (view instanceof ImageView))
52  			return true;
53  		else if(Issues.ISSUE_STATE.equals(colName) && (view instanceof ImageView))
54  			return true;
55  		else if(Issues.ISSUE_TYPE.equals(colName) && (view instanceof ImageView))
56  			((ImageView) view).setImageDrawable(getTypeIcon(
57  				cursor.getString(cursor.getColumnIndex(Issues.APP_NAME)), 
58  				cursor.getString(cursor.getColumnIndex(Issues.ISSUE_TYPE)), 
59  				cursor.getString(cursor.getColumnIndex(Issues.ISSUE_PRIORITY)), 
60  				cursor.getString(cursor.getColumnIndex(Issues.ISSUE_STATE))));
61  		else
62  			return false;
63  		
64  		return true;
65  	}
66  
67  	protected Drawable getTypeIcon(String appName, String type, String priority, String state)
68  	{
69  		IssueResource resource = factory.getRepository(appName).getState(type, priority, state);
70  		if(resource != null)
71  			return resource.getIcon(ctx);
72  		else
73  			return null;
74  	}
75  	
76  }
