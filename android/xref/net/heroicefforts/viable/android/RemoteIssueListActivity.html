41  /**
42   * This class handles the display of issues fetched from remote repositories.
43   * 
44   * @author jevans
45   *
46   */
47  public class RemoteIssueListActivity extends AbstractIssueListActivity
48  {
49      public static final int MENU_ITEM_REFRESH = Menu.FIRST;
50  	
51      private RepositoryFactory factory;
52  
53  	protected List<String> getAppNames()
54  	{
55  		List<String> appNames = new ArrayList<String>(getFactory().getApplicationNames());
56  		return appNames;
57  	}
58      
59  	protected Cursor getIssueCursor(String firstApp, String version) 
60  		throws ServiceException
61  	{
62  		RemoteIssueCursor cursor = new RemoteIssueCursor(getFactory().getRepository(firstApp));
63  		SearchParams params = new SearchParams();
64  		if(version != null)
65  			params.setProjectAffectedVersions(firstApp, Arrays.asList(new String[] { version }));
66  		else
67  			params.setProjectAffectedVersions(firstApp, Arrays.asList(new String[] { "all" }));
68  		cursor.setSearchParams(params);
69  
70  		return cursor;
71  	}
72  		 
73  	protected ArrayList<String> getVersionList(int position, String appName) 
74  		throws CreateException, ServiceException
75  	{
76  		List<VersionDetail> vDetails = factory.getRepository(appName).getApplicationVersions();
77  		ArrayList<String> versions = new ArrayList<String>();
78  		versions.add(getString(R.string.all));
79  		for(VersionDetail detail : vDetails)
80  			versions.add(detail.getName());
81  		return versions;
82  	}
83  	
84      protected OnItemClickListener getIssueListClickListener()
85      {
86      	return new OnItemClickListener()
87          {
88      		public void onItemClick(AdapterView<?> l, View v, int position, long id)
89      		{
90      	    	Cursor cursor = (Cursor) l.getAdapter().getItem(position);
91      	    	String appName = cursor.getString(cursor.getColumnIndex(Issues.APP_NAME));
92      	    	String issueId = cursor.getString(cursor.getColumnIndex(Issues.ISSUE_ID));
93      	    	Intent intent = new Intent(RemoteIssueListActivity.this, IssueViewActivity.class);
94      	    	intent.putExtra(IssueViewActivity.EXTRA_ISSUE_ID, issueId);
95      	    	intent.putExtra(IssueViewActivity.EXTRA_APP_NAME, appName);
96      	        startActivity(intent);
97      		}
98          };
99      }
100     
101     @Override
102     public boolean onCreateOptionsMenu(Menu menu) {
103         super.onCreateOptionsMenu(menu);
104 
105         menu.add(0, MENU_ITEM_REFRESH, 0, R.string.refresh)
106                 .setShortcut('3', 'r')
107                 .setIcon(android.R.drawable.ic_popup_sync);
108         
109         return true;
110     }	
111 	
112     @Override
113     public boolean onOptionsItemSelected(MenuItem item) {
114         switch (item.getItemId()) {
115         case MENU_ITEM_REFRESH:
116         	((SimpleCursorAdapter) listView.getAdapter()).getCursor().requery();
117         }
118         return super.onOptionsItemSelected(item);
119     }
120     
121     //late instantiation
122 	private RepositoryFactory getFactory()
123 	{
124 		if(factory == null)
125 			factory = new RepositoryFactory(this);
126 		
127 		return factory;
128 	}
