116     private OnItemSelectedListener appChosen = new OnItemSelectedListener()
117     {
118 		public void onItemSelected(AdapterView<?> l, View v, int position, long id)
119 		{
120 			String appName = (String) l.getItemAtPosition(position);
121 			if(appName.equals(getString(R.string.all)))
122 				appName = null;
123 
124 			new LoadIssuesTask().execute(String.valueOf(position), appName, null);
125 		}
126 
127 		public void onNothingSelected(AdapterView<?> arg0)
128 		{
129 			//empty
130 		}
131     	
132     };
133 
134     private OnItemSelectedListener versionChosen = new OnItemSelectedListener()
135     {
136 		public void onItemSelected(AdapterView<?> l, View v, int position, long id)
137 		{
138 			String appName = (String) appNameSpinner.getItemAtPosition(appNameSpinner.getLastVisiblePosition());
139 			String version = (String) l.getItemAtPosition(position);
140 			
141 			if(appName.equals(getString(R.string.all)))
142 				appName = null;
143 			if(position == 0)
144 				version = null;
145 			
146 			new LoadIssuesTask().execute(String.valueOf(-1), appName, version);
147 		}
148 
149 		public void onNothingSelected(AdapterView<?> arg0)
150 		{
151 			//empty
152 		}
153     	
154     };

156     /**
157      * Results class for returning asynchronous result values.
158      */
159 	private class IssueHolder
160 	{
161 		public List<String> versionList;
162 		public Cursor issueCursor;
163 		public ServiceException exc;
164 	}
165 
166 	/**
167 	 * Asynchronously loads the versions and issues for the specified application. 
168 	 */
169 	private class LoadIssuesTask extends AsyncTask<String, Void, IssueHolder> 
170 	{
171 		@Override
172 		protected void onPreExecute()
173 		{
174 			progressBar.setVisibility(View.VISIBLE);
175 		}
176 		
177 		/**
178 		 * @param position the application spinner position
179 		 * @param appName the application name
180 		 * @param version the version name
181 		 * @return a result with a list of version names and a list of issues.
182 		 */
183 		@Override
184 		protected IssueHolder doInBackground(String... params)
185 		{
186 			IssueHolder holder = new IssueHolder();
187 			try
188 			{
189 				int position = Integer.parseInt(params[0]);
190 				String appName = params[1];
191 				String version = params[2];
192 
193 				if(position >= 0)
194 					holder.versionList = getVersionList(position, appName);
195 				
196 				holder.issueCursor = getIssueCursor(appName, version);
197 			}
198 			catch (ServiceException e)
199 			{
200 				holder.exc = e;
201 			}			
202 				
203 			return holder;
204 		}
205 		
206 		@Override
207 		protected void onPostExecute(IssueHolder holder)
208 		{
209 			if(holder.exc == null)
210 			{
211 				if(holder.versionList != null)
212 				{
213 					ArrayAdapter<String> vAdapter = new ArrayAdapter<String>(AbstractIssueListActivity.this, android.R.layout.simple_spinner_item, holder.versionList);
214 					versionSpinner.setAdapter(vAdapter);
215 				}
216 				
217 				SimpleCursorAdapter adapter = new SimpleCursorAdapter(AbstractIssueListActivity.this, R.layout.issue_list_item, holder.issueCursor,
218 					    new String[] { Issues.SUMMARY, Issues.ISSUE_ID, Issues.ISSUE_TYPE, Issues.ISSUE_PRIORITY, Issues.ISSUE_STATE }, 
219 					    new int[] { R.id.SummaryTextView, R.id.IssueIdTextView, R.id.TypeImageView, R.id.PriorityImageView, R.id.StateImageView });
220 
221 				adapter.setViewBinder(new IssuesListViewBinder(AbstractIssueListActivity.this, factory));
222 				listView.setAdapter(adapter);				
223 			}
224 			else
225 			{
226 				Error.handle(AbstractIssueListActivity.this, holder.exc);
227 				SimpleCursorAdapter adapter = new SimpleCursorAdapter(AbstractIssueListActivity.this, R.layout.issue_list_item, new NullCursor(), new String[] {}, new int[] {});
228 				listView.setAdapter(adapter);								
229 			}
230 			
231 			progressBar.setVisibility(View.GONE);
232 		}
233 		
234 	}
