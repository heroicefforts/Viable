109 	 * 
110 	 * @param request the request to be executed.
111 	 * @return the client response
112 	 * @throws IOException if an error occurs
113 	 */
114 	private HttpResponse execute(HttpUriRequest request)
115 		throws IOException
116 	{
117 		if(Config.LOGV)
118 			Log.v(TAG, "Requesting:  " + request.getURI().toASCIIString());
119 		request.addHeader("Accept-Encoding", "gzip");
120 		if(authToken != null)
121 		{
122 			request.addHeader("Authorization", "GoogleLogin auth=" + authToken);
123 //			request.addHeader("Authorization", "AuthSub token=" + authToken);
124 			if(Config.LOGV)
125 				Log.v(TAG, "Added auth token '" + authToken + "' to request.");
126 		}
127 		return httpclient.execute(request);
128 	}

144 	/**
145 	 * Reads the response body and returns a string.  This also handles gzip decompression, if necessary.
146 	 * @param response the client response
147 	 * @return a string of the response body.
148 	 * @throws IOException if a service error occurs.
149 	 */
150 	private String readResponse(HttpResponse response) throws IOException
151 	{
152 		InputStream instream = response.getEntity().getContent();
153 		Header contentEncoding = response.getFirstHeader("Content-Encoding");
154 		if(Config.LOGV) //NOPMD
155 			if(contentEncoding != null) //NOPMD
156 				Log.v(TAG, "Response content encoding was '" + contentEncoding.getValue() + "'");
157 		if (contentEncoding != null && contentEncoding.getValue().equalsIgnoreCase("gzip")) {
158 			if(Config.LOGD)
159 				Log.d(TAG, "Handling GZIP response.");
160 		    instream = new GZIPInputStream(instream);
161 		}		
162 
163 		BufferedInputStream bis = new BufferedInputStream(instream);
164 		ByteArrayOutputStream baos = new ByteArrayOutputStream();
165 		byte[] buf = new byte[1024];
166 		int read = 0;
167 		while((read = bis.read(buf)) > 0)
168 			baos.write(buf, 0, read);
169 		String body = baos.toString();
170 		if(Config.LOGV)
171 			Log.v(TAG, "Response:  " + body);
172 		return body;
173 	}

267 	private IssueCommentsFeed loadCommentFeed(JSONObject obj)
268 		throws JSONException
269 	{
270 		if(Config.LOGV)
271 			Log.v(TAG, "CommentFeed query response:  " + obj.toString(4));
272 		List<Comment> comments = new ArrayList<Comment>();
273 		JSONObject feed = obj.getJSONObject("feed");
274 		if(feed.has(ENTRY))
275 		{
276 			JSONArray entries = feed.getJSONArray(ENTRY);
277 			for(int i = 0; i < entries.length(); i++)
278 				comments.add(loadComment(entries.getJSONObject(i)));
279 		}			
280 		
281 		int total = Integer.parseInt(feed.getJSONObject("openSearch$totalResults").getString("$t"));
282 		
283 		return new IssueCommentsFeed(comments, total);
284 	}

286 	private Comment loadComment(JSONObject obj)
287 		throws JSONException
288 	{
289 		String idStr = obj.getJSONObject("id").getString("$t");
290 		long id = Long.parseLong(idStr.substring(idStr.lastIndexOf('/') + 1));
291 		StringBuilder body = new StringBuilder();
292 		body.append(obj.getJSONObject("title").getString("$t")).append(EOL);
293 		body.append(EOL);
294 		body.append(obj.getJSONObject("content").getString("$t"));
295 		JSONArray authors = obj.getJSONArray("author");
296 		StringBuilder author = new StringBuilder();
297 		for(int i = 0; i < authors.length(); i++)
298 			author.append(authors.getJSONObject(i).getJSONObject("name").getString("$t")).append(", ");
299 		author.setLength(author.length() - 2);
300 
301 		Date created = null;
302 		try
303 		{
304 			created = sdf.parse(obj.getJSONObject(PUBLISHED_DATE).getString("$t"));
305 		}
306 		catch (ParseException e)
307 		{
308 			throw new JSONException("Error parsing JSON date:  " + obj.getJSONObject(PUBLISHED_DATE).getString("$t"));
309 		}
310 		
311 		return new Comment(id, body.toString(), author.toString(), created);
312 	}

315 	 * Inserts the supplied issue using the GData Issue protocol.
316 	 * @param postUrl the valid issue post URL.
317 	 * @param issue the issue to insert.
318 	 * @return the issue populated with the returned id and values.
319 	 * @throws AuthenticationException if authentication to the service fails
320 	 * @throws ServiceException if a service error occurs
321 	 */
322 	public Issue insert(URL postUrl, Issue issue)
323 		throws AuthenticationException, ServiceException
324 	{
325 		try
326 		{
327 			String appName = issue.getAppName();
328 			HttpPost post = new HttpPost(postUrl.toString());
329 			post.addHeader("Content-Type", "application/atom+xml");
330 			post.setEntity(new StringEntity(new AtomIssue(issue).toString()));
331 			HttpResponse response = execute(post);
332 			int responseCode = response.getStatusLine().getStatusCode();
333 			String body = readResponse(response);
334 			if(HttpStatus.SC_CREATED == responseCode)
335 			{
336 				Issue wireIssue = new AtomIssue(body);
337 				issue.copy(wireIssue);
338 			}
339 			else if(HttpStatus.SC_FORBIDDEN == responseCode || HttpStatus.SC_UNAUTHORIZED == responseCode)
340 				throw new AuthenticationException(responseCode, body);
341 			else
342 				throw new ServiceException("Exception creating issue:  " + body);
343 			
344 			issue.setAppName(appName);
345 			
346 			return issue;
347 		}
348 		catch (Exception e)
349 		{
350 			throw new ServiceException("Error creating issue.", e);
351 		}
352 	}

449 	private Issue loadIssue(JSONObject obj)
450 		throws JSONException
451 	{
452 		Issue issue = new Issue();
453 		issue.setSummary(obj.getJSONObject("title").getString("$t"));
454 		String bodyStr = obj.getJSONObject("content").getString("$t");
455 		int idxStart = bodyStr.indexOf(SALT_STACKTRACE);
456 		if (idxStart != -1)
457 		{
458 			int idxEnd = bodyStr.indexOf("</p>", idxStart);
459 			issue.setStacktrace(bodyStr.substring(idxStart + SALT_STACKTRACE.length(), idxEnd));
460 			issue.setDescription(bodyStr.substring(0, idxStart).replaceAll("<p>", "").replaceAll("</p>", EOL));
461 		}
462 		else
463 			issue.setDescription(bodyStr.replaceAll("<p>", "").replaceAll("</p>", EOL));
464 		
465 		issue.setIssueId(obj.getJSONObject("issues$id").getString("$t"));
466 		issue.setVotes(obj.getJSONObject("issues$stars").getLong("$t"));
467 		JSONArray labels = obj.getJSONArray("issues$label");
468 		ArrayList<String> affectedVersions = new ArrayList<String>();
469 		for(int i = 0; i < labels.length(); i++)
470 		{
471 			String label = labels.getJSONObject(i).getString("$t");
472 			if(Config.LOGV)
473 				Log.v(TAG, "Label:  " + label);
474 			if(label.startsWith("Type-"))
475 				issue.setType(label.substring("Type-".length()));
476 			else if(label.startsWith("Priority-"))
477 				issue.setPriority(label.substring("Priority-".length()));
478 			else if(label.startsWith("Hash-"))
479 				issue.setHash(label.substring("Hash-".length()));
480 			else if(label.startsWith("AffectedVersion-"))
481 				affectedVersions.add(label.substring("AffectedVersion-".length()));
482 		}
483 		issue.setAffectedVersions(affectedVersions.toArray(new String[affectedVersions.size()]));
484 		
485 		issue.setState(obj.getJSONObject("issues$state").getString("$t"));
486 		try
487 		{
488 			issue.setCreateDate(sdf.parse(obj.getJSONObject(PUBLISHED_DATE).getString("$t")));
489 		}
490 		catch (ParseException e)
491 		{
492 			throw new JSONException("Error parsing JSON date:  " + obj.getJSONObject(PUBLISHED_DATE).getString("$t"));
493 		}
494 
495 		try
496 		{
497 			issue.setModifiedDate(sdf.parse(obj.getJSONObject("updated").getString("$t")));
498 		}
499 		catch (ParseException e)
500 		{
501 			throw new JSONException("Error parsing JSON date:  " + obj.getJSONObject("updated").getString("$t"));
502 		}
503 		
504 		return issue;
505 	}
