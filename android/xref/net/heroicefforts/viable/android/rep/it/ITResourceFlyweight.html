82  	public static IssueResource getState(String type, String priority, String state)
83  	{
84  		IssueResource resource = ALL_STATES.get(type + priority + state);
85  		if(resource == null)
86  		{
87  			Log.w(TAG, "No issue state defined for Type:  '" + type + "', Priority:  '" + priority + "', State:  '" + state + "'.  Returning no-op.");
88  			resource = new NullIssueResource(type, priority, state);
89  		}
90  		
91  		return resource;
92  	}

170 	private static final void putDefault(HashMap<String, TypePriorityStateResource> states, HashSet<TypePriorityStateResource> def,
171 			String type, String priority, String state, int nameRes, int descRes, int iconRes)
172 	{
173 		def.add(put(states, type, priority, state, nameRes, descRes, iconRes));
174 	}
175 	
176 	private static final TypePriorityStateResource put(HashMap<String, TypePriorityStateResource> states, String type, String priority,
177 			String state, int nameRes, int descRes, int iconRes)
178 	{
179 		ITIssueResource resource = new ITIssueResource(type, priority, state, nameRes, descRes, iconRes);
180 		for(String t : select(type, TYPES))
181 			for(String p : select(priority, PRIORITIES))
182 				for(String s : select(state, STATES))
183 					states.put(t + p + s, resource);
184 
185 		return resource;
186 	}
187 
188 	private static final String[] select(String type, String[] all)
189 	{
190 		if(ANY.equals(type))
191 			return all;
192 		else
193 			return new String[] { type };
194 	}
