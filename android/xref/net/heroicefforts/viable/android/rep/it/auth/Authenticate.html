52  /**
53   * This class is used to interact with the Google Client Login protocol to attain user tokens for Google services.
54   * 
55   * @author jevans
56   *
57   */
58  public class Authenticate
59  {
60  	private static final String TAG = "Authenticate";
61  
62  	/**
63  	 * Attempts to authenticate the user using a pre-existing stored authentication token.  If an account exists, but no such token 
64  	 * exists, then the user will be prompted by the account authenticator to re-enter their Google credentials to generate the new token.
65  	 * 
66  	 * @param act the calling activity
67  	 * @return the authentication token for the requested service or null if there is no Google Account.
68  	 * @throws AuthenticatorException if an error occurs during authentication.
69  	 * @throws OperationCanceledException
70  	 * @throws IOException
71  	 */
72  	public static String authenticate(Activity act, String serviceCode)
73  		throws AuthenticatorException, OperationCanceledException, IOException
74  	{
75  		AccountManager mgr = AccountManager.get(act); 
76          Account[] accts = mgr.getAccountsByType(GCLAccountAuthenticator.ACCT_TYPE); 
77          if(accts.length > 0)
78          {
79  	        Account acct = accts[0]; 
80  	        AccountManagerFuture<Bundle> accountManagerFuture = mgr.getAuthToken(acct, serviceCode, null, act, null, null); 
81  	        Bundle authTokenBundle = accountManagerFuture.getResult(); 
82  	        String authToken = authTokenBundle.get(AccountManager.KEY_AUTHTOKEN).toString();
83  	        
84  	        return authToken; 
85          }
86          else
87          {
88          	Log.e(TAG, "No google accounts registered for this device.");
89          	return null;
90          }
91  	}

93  	/**
94  	 * Directly posts to Google Client Login service to generate an authentication token.
95  	 * 
96  	 * @param username
97  	 * @param password
98  	 * @param service the code for the service desired e.g. code
99  	 * @return the authentication token for that service.
100 	 * @throws AuthenticationException if authentication failed.
101 	 * @throws NetworkException if there is an error communicating with the login service.
102 	 */
103 	public static final String authenticate(String username, String password, String service) 
104 		throws AuthenticationException, NetworkException
105 	{
106 		HttpClient httpclient = new DefaultHttpClient();
107 		HttpPost post = new HttpPost("https://www.google.com/accounts/ClientLogin");
108 		List<NameValuePair> nameValuePairs = new ArrayList<NameValuePair>(12);
109 		nameValuePairs.add(new BasicNameValuePair("accountType", "HOSTED_OR_GOOGLE"));
110 		nameValuePairs.add(new BasicNameValuePair("Email", username));
111 		nameValuePairs.add(new BasicNameValuePair("Passwd", password));
112 		nameValuePairs.add(new BasicNameValuePair("service", service));
113 		nameValuePairs.add(new BasicNameValuePair("source", "heroicefforts-viable-1.0.0"));
114 		
115 		int responseCode;
116 		String body;
117 		try
118 		{
119 			post.setEntity(new UrlEncodedFormEntity(nameValuePairs));  
120 			
121 			HttpResponse response = httpclient.execute(post);
122 			
123 			responseCode = response.getStatusLine().getStatusCode();
124 			body = readResponse(response);
125 		}
126 		catch (Exception e)
127 		{
128 			throw new NetworkException("Exception during authentication.", e);
129 		}
130 		
131 		if(HttpStatus.SC_OK == responseCode)
132 		{	
133 			if(Config.LOGV)
134 				Log.v(TAG, "Auth body response:  " + body);
135 			Pattern authPat = Pattern.compile("Auth=([A-Za-z0-9_-]+)");
136 			Matcher m = authPat.matcher(body);
137 			if(m.find())
138 			{
139 				String authToken = m.group(1);
140 				return authToken;
141 			}
142 			else
143 				throw new AuthenticationException(responseCode, "Couldn't locate Auth token in login response.");
144 		}		
145 		else
146 			throw new AuthenticationException(responseCode, "Authentication failed:  " + body);
147 	}
148 	
149 	private static String readResponse(HttpResponse response) throws IOException
150 	{
151 		InputStream instream = response.getEntity().getContent();
152 		Header contentEncoding = response.getFirstHeader("Content-Encoding");
153 		if(Config.LOGV)	//NOPMD
154 			if(contentEncoding != null) //NOPMD
155 				Log.v(TAG, "Response content encoding was '" + contentEncoding.getValue() + "'");
156 		if (contentEncoding != null && contentEncoding.getValue().equalsIgnoreCase("gzip")) 
157 		{
158 			if(Config.LOGD)
159 				Log.d(TAG, "Handling GZIP response.");
160 		    instream = new GZIPInputStream(instream);
161 		}		
162 
163 		BufferedInputStream bis = new BufferedInputStream(instream);
164 		ByteArrayOutputStream baos = new ByteArrayOutputStream();
165 		byte[] buf = new byte[1024];
166 		int read = 0;
167 		while((read = bis.read(buf)) > 0)
168 			baos.write(buf, 0, read);
169 		String body = baos.toString();
170 		if(Config.LOGV)
171 			Log.v(TAG, "Response:  " + body);
172 		return body;
173 	}
