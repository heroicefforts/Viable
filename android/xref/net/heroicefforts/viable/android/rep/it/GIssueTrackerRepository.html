117 	public GIssueTrackerRepository(String appName, Activity act, Bundle metaData)
118 		throws CreateException
119 	{
120 		this.appName = appName;
121 		this.act = act;
122 		this.projectName = metaData.getString(PARAM_PROJECT_NAME);
123 		if(projectName == null)
124 			throw new CreateException("No '" + "viable-project-name" + "' meta-data field defined for application.  Google Isusue Tracker Repository cannot be constructed.");
125 		this.description = metaData.getString(PARAM_PROJECT_DESC);
126 		this.lead = metaData.getString(PARAM_PROJECT_ADMIN);
127 		String versionStr = metaData.getString(PARAM_VERSIONS);
128 		if(versionStr != null)
129 		{
130 			String[] versions = versionStr.split("[ ]*,[ ]*");
131 			for(String version : versions)
132 				this.versions.add(new VersionDetail(version));
133 		}
134 		
135 		try
136 		{				
137 			host = new ProjectHostingService();
138 			String token = Authenticate.authenticate(act, GCLAccountAuthenticator.TOKEN_TYPE_ISSUE_TRACKER);
139 			if(token != null)
140 				host.setAuthSubToken(token);
141 			else
142 			{
143 				//TODO add popup.
144 				AccountManager acct = (AccountManager) act.getSystemService(Context.ACCOUNT_SERVICE);
145 				if(Config.LOGD)
146 					Log.d(TAG, "Requesting account creation.");
147 				acct.addAccount(GCLAccountAuthenticator.ACCT_TYPE, "code", null, null, act, callback  , null);
148 			}
149 		}
150 		catch (AuthenticatorException e)
151 		{
152 			throw new CreateException("Exception authenticating Google account for Issue Tracker repository access.");
153 		}
154 		catch (Exception e)
155 		{
156 			throw new CreateException("Exception creating Google Issue Tracker repository.", e);
157 		}
158 	}

175 	private Issue findByHash(String hash) 
176 		throws ServiceException
177 	{
178 		try
179 		{
180 			Issue retVal = null;
181 
182 			if(hash != null)
183 			{
184 				URL feedUrl = new URL(ISSUE_BASE_URL + projectName + "/issues/full");
185 				IssuesQuery myQuery = new IssuesQuery(feedUrl);
186 				myQuery.setLabel("Hash-" + hash);
187 				IssuesFeed resultFeed = host.query(myQuery);
188 	
189 				if (resultFeed.getEntries().size() > 0)
190 				{
191 					retVal = resultFeed.getEntries().get(0);
192 					retVal.setAppName(appName);
193 				}
194 			}
195 			
196 			return retVal;
197 		}
198 		catch (MalformedURLException e)
199 		{
200 			throw new ServiceException("Error retrieving issue by hash '" + hash + "'.", e);
201 		}

223 		throws ServiceException
224 	{
225 		try
226 		{
227 			int startIndex = ((page - 1) * pageSize) + 1; // 1 based
228 			int maxResults = pageSize + 1;
229 			URL feedUrl = new URL(ISSUE_BASE_URL + projectName + "/issues/" + issueId
230 					+ "/comments/full" + "?start-index=" + startIndex + APPEND_MAX_RESULTS + maxResults);
231 			IssueCommentsFeed resultFeed = host.getFeed(feedUrl);
232 			List<Comment> comments = resultFeed.getEntries();
233 			CommentSet set = null;
234 			if (comments.size() > pageSize)
235 				set = new CommentSet(comments.subList(0, pageSize), true);
236 			else
237 				set = new CommentSet(comments, false);
238 
239 			return set;
240 		}
241 		catch (MalformedURLException e)
242 		{
243 			throw new ServiceException("Error retrieving comments for issue '" + issueId + "'.", e);
244 		}

344 	{

346 		if(Config.LOGD)

382 	private String createHash(Issue issue)
383 	{
384 		String hash = null;
385 
386 		if (issue.getStacktrace() != null)
387 		{
388 			String projectKey = projectName;
389 			try
390 			{
391 				MessageDigest md = MessageDigest.getInstance("SHA");
392 
393 				try
394 				{
395 					md.update(projectKey.getBytes("UTF8"));
396 					md.update(issue.getStacktrace().getBytes("UTF8"));
397 					byte[] hashBytes = md.digest();
398 					String hex = new BigInteger(1, hashBytes).toString(16);
399 					if (hex.length() % 2 != 0)
400 						hex = "0" + hex;
401 					hash = hex;
402 					issue.setHash(hash);
403 				} 
404 				catch (UnsupportedEncodingException e)
405 				{
406 					Log.e(TAG, "Error generating bug hash using UTF8 encoding.  Hash disabled.", e);
407 				}
408 			}
409 			catch (NoSuchAlgorithmException e)
410 			{
411 				Log.e(TAG, "Error generating bug hash.  Failed to find SHA digest.  Hash disabled.", e);
412 			}
413 		}
414 
415 		return hash;
416 	}

418 	public SearchResults search(SearchParams params) 
419 		throws ServiceException
420 	{
421 		final int startIdx = (params.getPage() - 1) * params.getPageSize();
422 		final int pageSize = params.getPageSize();
423 		
424 		LinkedHashSet<Issue> issues = new LinkedHashSet<Issue>();		
425 		for(int i = startIdx; i < params.getIds().size() && i < startIdx + pageSize + i; i++)
426 			issues.add(findById(params.getIds().get(i)));
427 		
428 		if(issues.size() <= pageSize)
429 		{
430 			Issue issue = findByHash(params.getHash());
431 			if(issue != null)
432 				issues.add(issue);
433 		}	
434 		
435 		if(issues.size() <= pageSize)
436 		{
437 			long skip = startIdx;
438 			long skipped = params.getIds().size() + (params.getHash() != null ? 1 : 0);
439 			for(int i = 0; i < params.getAffectedVersions().size() && issues.size() <= pageSize; i++)
440 			{
441 				long count = countIssuesByVersion(params.getAffectedVersions().get(i));
442 				if(skipped + count > skip)
443 				{
444 					int needed = pageSize - issues.size() + 1;
445 					long startAt = skip - skipped;
446 					if(startAt < 0)
447 						startAt = 0;
448 					issues.addAll(findIssuesByVersion(params.getAffectedVersions().get(i), startAt, needed));
449 					skipped = skip;
450 				}
451 				else
452 				{
453 					skipped += count;
454 				}
455 			}
456 		}
457 		
458 		setAppName(issues);
459 		
460 		if(issues.size() > pageSize)
461 			return new SearchResults(new ArrayList<Issue>(issues).subList(0, pageSize), true);
462 		else
463 			return new SearchResults(new ArrayList<Issue>(issues), false);
464 	}
