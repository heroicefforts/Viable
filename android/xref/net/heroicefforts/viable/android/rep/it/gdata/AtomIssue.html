40  /**
41   * This class extends issue to add handling of the Gdata Atom protocol.
42   * 
43   * @author jevans
44   *
45   */
46  public class AtomIssue extends Issue
47  {
48  	private static final String TAG = "AtomIssue";
49  
50  	private static final String BOP = "<p>";
51  	private static final String EOP = "</p>";
52  	private static final String SALT_STACKTRACE = "<p>Stacktrace:  ";
53  
54  	private static final String TYPE_DEFECT = "Defect";
55  
56  	private static final String LABEL_AFFECTED_VERSION = "AffectedVersion";
57  	private static final String LABEL_DEVICE = "Device";
58  	private static final String LABEL_HASH = "Hash";
59  	private static final String LABEL_MODEL = "Model";
60  	private static final String LABEL_PRIORITY = "Priority";
61  	private static final String LABEL_SDK = "SDK";
62  	private static final String LABEL_TYPE = "Type";
63  
64  	
65  	
66  	public AtomIssue(Issue issue)
67  	{
68  		this.copy(issue);
69  	}
70  
71  	public AtomIssue(String atomEntry)
72  		throws SAXException, IOException, ParserConfigurationException
73  	{
74  		parse(atomEntry);
75  	}
76  	
77  	protected void parse(String atomEntry)
78  		throws SAXException, IOException, ParserConfigurationException
79  	{
80  		if(Config.LOGV)
81  			Log.v(TAG, "Parsing:  " + atomEntry);
82  		SAXParserFactory f = SAXParserFactory.newInstance();
83  		SAXParser p = f.newSAXParser();
84  		p.parse(new InputSource(new StringReader(atomEntry)), new IssueContentHandler(this));
85  	}
86  	
87  	public String toString()
88  	{
89  		StringBuilder builder = new StringBuilder();
90  		builder.append("<?xml version='1.0' encoding='UTF-8'?>");
91  		builder.append("<entry xmlns='http://www.w3.org/2005/Atom' xmlns:issues='http://schemas.google.com/projecthosting/issues/2009'>");
92  		builder.append("<title>").append(getSummary()).append("</title>");
93  		builder.append("<content type='html'>").append(getBody()).append("</content>");
94  		builder.append("<author><name>Anonymous</name></author>");
95  		builder.append("<issues:stars>").append(getVotes()).append("</issues:stars>");
96  		builder.append(getLabels());
97  		builder.append("</entry>");
98  
99  		return builder.toString();
100 	}

102 	private Object getLabels()
103 	{
104 		StringBuilder labels = new StringBuilder();
105 		appendLabel(labels, LABEL_TYPE, getType());
106 		appendLabel(labels, LABEL_PRIORITY, getPriority());
107 		appendLabel(labels, LABEL_HASH, getHash());
108 		for(String version : getAffectedVersions())
109 			appendLabel(labels, LABEL_AFFECTED_VERSION, version);
110 		if(TYPE_DEFECT.equals(getType()))
111 		{
112 			appendLabel(labels, LABEL_MODEL, Build.MODEL);
113 			appendLabel(labels, LABEL_DEVICE, Build.DEVICE);
114 			appendLabel(labels, LABEL_SDK, String.valueOf(Build.VERSION.SDK_INT));
115 		}
116 				
117 		return labels;
118 	}
119 
120 	private void appendLabel(StringBuilder labels, String type, String value)
121 	{
122 		if(TextUtils.isEmpty(value))
123 			labels.append("<issues:label>").append(type + "-" + value).append("</issues:label>");
124 	}
125 
126 	private StringBuilder getBody()
127 	{
128 		StringBuilder body = new StringBuilder();
129 		body.append("<![CDATA[<p>").append(getDescription()).append(EOP);
130 		body.append(BOP).append("Affected Versions:  ").append(Arrays.asList(getAffectedVersions())).append(EOP);
131 		if (getStacktrace() != null)
132 			body.append(SALT_STACKTRACE).append(getStacktrace()).append(EOP);
133 		body.append("]]>");
134 		return body;
135 	}

137 	private static class IssueContentHandler extends AtomHandler
138 	{
139 		private Issue issue;
140 		private ArrayList<String> affectedVersions = new ArrayList<String>();
141 		private ArrayList<String> affectedDevices = new ArrayList<String>();
142 		private ArrayList<String> affectedModels = new ArrayList<String>();
143 		private ArrayList<String> affectedSDKs = new ArrayList<String>();
144 
145 		
146 		public IssueContentHandler(Issue issue)
147 		{
148 			this.issue = issue;
149 		}
150 		
151 		@Override
152 		public void endElement(String uri, String localName, String qName) throws SAXException
153 		{
154 			super.endElement(uri, localName, qName);
155 			
156 			if("id".equals(localName))
157 			{
158 				issue.setIssueId(value.substring(value.lastIndexOf("/") + 1));
159 			}
160 			else if("title".equals(localName))
161 			{
162 				issue.setSummary(value.toString());
163 			}
164 			else if("content".equals(localName))
165 			{
166 				int idxStart = value.indexOf(SALT_STACKTRACE);
167 				if (idxStart != -1)
168 				{
169 					int idxEnd = value.indexOf(EOP, idxStart);
170 					issue.setStacktrace(value.substring(idxStart + SALT_STACKTRACE.length(), idxEnd));
171 					issue.setDescription(value.substring(0, idxStart).replaceAll(BOP, "").replaceAll(EOP, EOL));
172 				}
173 				else
174 					issue.setDescription(value.toString().replaceAll(BOP, "").replaceAll(EOP, EOL));
175 			}
176 			else if("published".equals(localName))
177 			{
178 				issue.setCreateDate(parseDate(value));
179 			}
180 			else if("updated".equals(localName))
181 			{
182 				issue.setModifiedDate(parseDate(value));
183 			}
184 			else if("label".equals(localName))
185 			{
186 				int idx = value.indexOf("-");
187 				if(idx != -1)
188 				{
189 					String type = value.substring(0, idx);
190 					String val = value.substring(idx + 1);
191 					
192 					if(LABEL_TYPE.equals(type))
193 						issue.setType(val);
194 					else if(LABEL_PRIORITY.equals(type))
195 						issue.setPriority(val);
196 					else if(LABEL_HASH.equals(type))
197 						issue.setHash(val);
198 					else if(LABEL_AFFECTED_VERSION.equals(type))
199 						affectedVersions.add(val);
200 					else if(LABEL_DEVICE.equals(type))
201 						affectedDevices.add(val);
202 					else if(LABEL_MODEL.equals(type))
203 						affectedModels.add(val);
204 					else if(LABEL_SDK.equals(type))
205 						affectedSDKs.add(val);					
206 				}
207 			}
208 			else if("state".equals(localName))
209 			{
210 				issue.setState(value.toString());
211 			}
212 		}
213 	
214 		@Override
215 		public void endDocument() throws SAXException
216 		{
217 			issue.setAffectedVersions(affectedVersions.toArray(new String[affectedVersions.size()]));
218 			issue.setAffectedDevices(affectedDevices.toArray(new String[affectedDevices.size()]));
219 			issue.setAffectedModels(affectedModels.toArray(new String[affectedModels.size()]));
220 			issue.setAffectedSDKs(affectedSDKs.toArray(new String[affectedSDKs.size()]));
221 		}
222 	
223 	}
