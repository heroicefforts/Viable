72  	public static IssueResource getState(String type, String priority, String state)
73  	{
74  		IssueResource resource = ALL_STATES.get(type + priority + state);
75  		if(resource == null)
76  		{
77  			Log.w(TAG, "No issue state defined for Type:  '" + type + "', Priority:  '" + priority + "', State:  '" + state + "'.  Returning no-op.");
78  			resource = new NullIssueResource(type, priority, state);
79  		}
80  		
81  		return resource;
82  	}

178 	private static final void putDefault(HashMap<String, TypePriorityStateResource> states, HashSet<TypePriorityStateResource> def,
179 			String type, String priority, String state, int nameRes, int descRes, int iconRes)
180 	{
181 		def.add(put(states, type, priority, state, nameRes, descRes, iconRes));
182 	}
183 	
184 	private static final TypePriorityStateResource put(HashMap<String, TypePriorityStateResource> states, String type, String priority,
185 			String state, int nameRes, int descRes, int iconRes)
186 	{
187 		JIRAIssueResource resource = new JIRAIssueResource(type, priority, state, nameRes, descRes, iconRes);
188 		for(String t : select(type, TYPES))
189 			for(String p : select(priority, PRIORITIES))
190 				for(String s : select(state, STATES))
191 					states.put(t + p + s, resource);
192 
193 		return resource;
194 	}
195 
196 	private static final String[] select(String type, String[] all)
197 	{
198 		if(ANY.equals(type))
199 			return all;
200 		else
201 			return new String[] { type };
202 	}
