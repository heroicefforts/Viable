74  /**
75   * This is the repository implementation that bridges to the JIRA Issue Tracker.  Viable client applications that
76   * wish to register their JIRA project should add lines similar to those found below:<br/>
77   * <br/><pre>
78     &lt;meta-data android:name="viable-provider" android:value="JIRA" /&gt;
79     &lt;meta-data android:name="viable-provider-location" android:value="http://&lt;your domain&gt;/rest/viable/1.0/issue" /&gt;	
80   * </pre><br/>
81   *
82   * Please note that you must install Viable's server-side JIRA plugin before the Viable client can connect.  See 
83   * the viable-jira-plugin subproject for further details.
84   * 
85   * @author jevans

110     private JIRARepository(String appName, String location)
111     	throws CreateException
112 	{
113 		if(location == null)
114 			throw new CreateException("No '" + "viable-provider-location" + "' meta-data field defined for application.  JIRA Repository cannot be constructed.");
115 
116 		this.rootURL = location;
117 		this.appName = appName;
118 		
119 		DefaultHttpClient client = new DefaultHttpClient();
120 		SchemeRegistry schemes = client.getConnectionManager().getSchemeRegistry();		
121 		BasicHttpParams params = new BasicHttpParams();
122 		HttpConnectionParams.setConnectionTimeout(params, CONN_TIMEOUT);
123 		httpclient = new DefaultHttpClient(new ThreadSafeClientConnManager(params, schemes), params);
124 	}

184     public CommentSet findCommentsForIssue(String issueId, int page, int pageSize)
185     	throws ServiceException
186     {
187 		try
188 		{
189 			StringBuilder url = new StringBuilder();
190 			url.append(rootURL).append("/issue/").append(issueId).append("/comments");
191 			if(page > 0)
192 				url.append("?page=").append(page).append("&pageSize=").append(pageSize);
193 
194 			HttpGet get = new HttpGet(url.toString());			
195 			HttpResponse response = execute(get);
196 			int code = response.getStatusLine().getStatusCode();
197 			if(HttpStatus.SC_NOT_FOUND == code)
198 			{
199 				if(Config.LOGD)
200 					Log.d(TAG, "No comments for issue '" + issueId + "'.");
201 				List<Comment> comments = Collections.emptyList();
202 				return new CommentSet(comments, false);
203 			}
204 			else if(HttpStatus.SC_OK == code)
205 			{
206 	        	JSONObject obj = readJSON(response);
207 	    		if(Config.LOGV)
208 	    			Log.v(TAG, "Comments JSON:  " + obj.toString(4));
209 	        	List<Comment> issues = loadComments(obj);
210 	        	boolean more = obj.getBoolean("more");
211 	    		if(Config.LOGD)
212 	    			Log.d(TAG, "Searched returned " + issues.size() + " results.");
213 				return new CommentSet(issues, more);
214 			}
215 			else
216 				throw new ServiceException(MSG_REMOTE_ERROR + code);
217 		}
218 		catch (JSONException e)
219 		{
220 			throw new ServiceException(MSG_PARSE_ERROR, e);
221 		}
222 		catch (IOException e)
223 		{
224 			throw new ServiceException(MSG_CONNECT_ERROR, e);
225 		}

232     public SearchResults search(SearchParams params) 
233     	throws ServiceException
234     {
235 		try
236 		{
237 			HttpPost post = new HttpPost(rootURL + "/search");		
238 			List<NameValuePair> nameValuePairs = new ArrayList<NameValuePair>(1);
239 			
240 			if(params.getHash() != null)
241 				nameValuePairs.add(new BasicNameValuePair("hash", params.getHash()));
242 			
243 			if(params.getIds() != null)
244 				for(String id : params.getIds())
245 					nameValuePairs.add(new BasicNameValuePair("issue_id", id));
246 			
247 			if(params.getAffectedVersions() != null)
248 				for(String version : params.getAffectedVersions())
249 					nameValuePairs.add(new BasicNameValuePair("affected_version", version));
250 			
251 			if(params.getAppName() != null)
252 				nameValuePairs.add(new BasicNameValuePair("app_name", params.getAppName()));
253 			
254 			if(params.getPage() > 0)
255 				nameValuePairs.add(new BasicNameValuePair("page", String.valueOf(params.getPage())));
256 			
257 			if(params.getPageSize() > 0)
258 				nameValuePairs.add(new BasicNameValuePair("page_size", String.valueOf(params.getPageSize())));
259 			
260 			if(Config.LOGD)
261 				Log.d(TAG, "Posting search paramters:  " + nameValuePairs.toString());
262 			
263 			post.setEntity(new UrlEncodedFormEntity(nameValuePairs));  		   
264 			HttpResponse response = execute(post);
265 			int code = response.getStatusLine().getStatusCode();
266 			if(HttpStatus.SC_NOT_FOUND == code)
267 			{
268 				if(Config.LOGD)
269 					Log.d(TAG, "Search returned no results:  '" + params + "'");
270 				List<Issue> empty = Collections.emptyList();
271 				return new SearchResults(empty, false);
272 			}
273 			else if(HttpStatus.SC_OK == code)
274 			{
275 	        	JSONObject obj = readJSON(response);
276 	    		if(Config.LOGV)
277 	    			Log.v(TAG, "Search JSON:  " + obj.toString(4));
278 	        	List<Issue> issues = loadIssues(obj);
279 	        	boolean more = obj.getBoolean("more");
280 	    		if(Config.LOGD)
281 	    			Log.d(TAG, "Searched returned " + issues.size() + " results:  " + params);
282 				return new SearchResults(issues, more);
283 			}
284 			else
285 				throw new ServiceException(MSG_REMOTE_ERROR + code);
286 		}
287 		catch (JSONException e)
288 		{
289 			throw new ServiceException(MSG_PARSE_ERROR, e);
290 		}
291 		catch (IOException e)
292 		{
293 			throw new ServiceException(MSG_CONNECT_ERROR, e);
294 		}

411 	public int postIssue(Issue issue) 
412 		throws ServiceException
413 	{
414 		try
415 		{
416 			HttpPost post = new HttpPost(rootURL + "/issue.json");
417 			List<NameValuePair> nameValuePairs = new ArrayList<NameValuePair>(12);
418 			nameValuePairs.add(new BasicNameValuePair("issue_type", issue.getType()));
419 			nameValuePairs.add(new BasicNameValuePair("priority", issue.getPriority()));				        
420 			nameValuePairs.add(new BasicNameValuePair("app_name", issue.getAppName()));
421 			nameValuePairs.add(new BasicNameValuePair("summary", issue.getSummary()));				        
422 			nameValuePairs.add(new BasicNameValuePair("description", issue.getDescription()));
423 			for(String version : issue.getAffectedVersions())
424 				nameValuePairs.add(new BasicNameValuePair("app_version_name", version));
425 			if(issue.getStacktrace() != null)
426 				nameValuePairs.add(new BasicNameValuePair("stacktrace", issue.getStacktrace()));
427 
428 			if(issue instanceof BugContext)
429 			{
430 				nameValuePairs.add(new BasicNameValuePair("phone_model", Build.MODEL));
431 				nameValuePairs.add(new BasicNameValuePair("phone_device", Build.DEVICE));				        
432 				nameValuePairs.add(new BasicNameValuePair("phone_version", String.valueOf(Build.VERSION.SDK_INT)));
433 			}
434 
435 			if(Config.LOGD)
436 				Log.d(TAG, "post params:  " + nameValuePairs.toString());
437 			post.setEntity(new UrlEncodedFormEntity(nameValuePairs));  
438 
439 			// Execute HTTP Post Request  
440 			HttpResponse response = execute(post);
441 			int responseCode = response.getStatusLine().getStatusCode();
442 			if(HttpStatus.SC_OK == responseCode || HttpStatus.SC_CREATED == responseCode)
443 			{			
444 				JSONObject obj = readJSON(response);
445 				if(Config.LOGV)
446 					Log.v(TAG, "postIssue response:  \n" + obj.toString(4));		
447 				//TODO pull dates from response
448 				Issue newIssue = new Issue(obj.getJSONObject("issue").toString(4));
449 				issue.copy(newIssue);
450 			}
451 			else
452 				throw new ServiceException(MSG_REMOTE_ERROR + responseCode);
453 			
454 			if(Config.LOGD)
455 				Log.d(TAG, "postIssue code " + responseCode);
456 			
457 			return responseCode;
458 		}
459 		catch (UnsupportedEncodingException e)
460 		{
461 			throw new ServiceException("Encoding error occurred parsing JIRA response", e);
462 		}
463 		catch (IOException e)
464 		{
465 			throw new ServiceException(MSG_CONNECT_ERROR, e);
466 		}
467 		catch (JSONException e)
468 		{
469 			throw new ServiceException(MSG_PARSE_ERROR, e);
470 		}
471 	}

474 	private String createHash(Issue issue)
475 	{
476 		String hash = null;
477 		try
478 		{
479 			MessageDigest md = MessageDigest.getInstance("SHA");
480 
481 			try
482 			{
483 				md.update(issue.getAppName().getBytes("UTF8"));
484 				md.update(issue.getStacktrace().getBytes("UTF8"));
485 				byte[] hashBytes = md.digest();
486 				String hex = new BigInteger(1, hashBytes).toString(16);
487 				if(hex.length() % 2 != 0)
488 					hex = "0" + hex;
489 				hash = hex;
490 			}
491 			catch (UnsupportedEncodingException e)
492 			{
493 				Log.e(TAG, "Error generating bug hash using UTF8 encoding.", e);
494 			}
495 		}
496 		catch (NoSuchAlgorithmException e)
497 		{
498 			Log.e(TAG, "Error generating bug hash.  Failed to find SHA digest.", e);
499 		}
500 		return hash;
501 	}

510 	private JSONObject readJSON(HttpResponse response) 
511 		throws IOException, JSONException
512 	{
513 		InputStream instream = response.getEntity().getContent();
514 		Header contentEncoding = response.getFirstHeader("Content-Encoding");
515 		if(Config.LOGV) //NOPMD
516 			if(contentEncoding != null) //NOPMD
517 				Log.v(TAG, "Response content encoding was '" + contentEncoding.getValue() + "'");
518 
519 		if (contentEncoding != null && contentEncoding.getValue().equalsIgnoreCase("gzip")) {
520 			if(Config.LOGD)
521 				Log.d(TAG, "Handling GZIP response.");
522 		    instream = new GZIPInputStream(instream);
523 		}		
524 
525 		BufferedInputStream bis = new BufferedInputStream(instream);
526 		ByteArrayOutputStream baos = new ByteArrayOutputStream();
527 		byte[] buf = new byte[1024];
528 		int read = 0;
529 		while((read = bis.read(buf)) > 0)
530 			baos.write(buf, 0, read);
531 		
532 		JSONObject obj = new JSONObject(baos.toString());
533 		return obj;
534 	}
