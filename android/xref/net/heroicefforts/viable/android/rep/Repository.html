32  /**
33   * This is the main service interface implemented by bridges to the various issue tracking systems.
34   * 
35   * @author jevans
36   *
37   */
38  public interface Repository
39  {
40  	/**
41  	 * The service performs a checksum lookup to determine if the issue data already exists in the repository.  If so,
42  	 * then the issue is returned.  Checksum calcs are to be determined by the service.
43  	 * @param issue the issue to check
44  	 * @return the located issue, or null
45  	 * @throws ServiceException if a service error occurs.
46  	 */
47  	public abstract Issue exists(Issue issue)
48  		throws ServiceException;
49  
50  	/**
51  	 * The service attempts to locate the issue by unique id.
52  	 * @param issueId the issue's unique identifier as provided by the repository.
53  	 * @return the located issue, or null
54  	 * @throws ServiceException if a service error occurs.
55  	 */
56  	public abstract Issue findById(String issueId) 
57  		throws ServiceException;
58  
59  	/**
60  	 * Searches for issues matching the supplied parameters.
61  	 * @param params the search criteria
62  	 * @return results matching the critieria.
63  	 * @throws ServiceException if a service error occurs.
64  	 */
65  	public abstract SearchResults search(SearchParams params)
66  		throws ServiceException;
67  
68  	/**
69  	 * Adds a new issue to the repository.  The supplied issue should be updated with the repository supplied id
70  	 * and any other changed data.
71  	 * @param issue the issue to add
72  	 * @return the response code
73  	 * @throws ServiceException if a service error occurs.
74  	 */
75  	public abstract int postIssue(Issue issue) 
76  		throws ServiceException;
77  
78  	/**
79  	 * Adds a new issue comment to the repository.  The supplied comment should be updated with the repository supplied id
80  	 * and any other changed data.
81  	 * @param issue the issue to which the comment should be posted
82  	 * @param comment the comment to post
83  	 * @return the response code
84  	 * @throws ServiceException if a service error occurs.
85  	 */
86  	public int postIssueComment(Issue issue, Comment comment) 
87  		throws ServiceException;
88  	
89  	/**
90  	 * Returns a page of comments belonging to the issue specified.
91  	 * @param issueId the id of the issue
92  	 * @param page the page to fetch (1-based).
93  	 * @param pageSize the number of results to return.
94  	 * @return the page of comments
95  	 * @throws ServiceException if a service error occurs.
96  	 */
97      public CommentSet findCommentsForIssue(String issueId, int page, int pageSize) 
98      	throws ServiceException;
99      
100     /**
101      * Returns summary data corresponding to this repository / project.  Some of this data may be pulled from the repository
102      * or from the client's application meta-data depending upon the implementation limitations.
103      * 
104      * @return the project summary details
105      * @throws ServiceException if a service error occurs.
106      */
107     public ProjectDetail getApplicationStats()
108     	throws ServiceException;
109 
110     /**
111      * Returns the release version details associated with this repository / project.
112      * @return a non-null list of release details.
113      * @throws ServiceException if a service error occurs.
114      */
115     public List<VersionDetail> getApplicationVersions()
116 		throws ServiceException;
117 
118     /**
119      * Submits a vote for the specified issue on behalf of the user.
120      * @param issue the issue for which to vote
121      * @return true if the vote was successful.
122      * @throws ServiceException
123      */
124 	public boolean vote(Issue issue)
125 		throws ServiceException;
126     
127     /**
128      * Returns an IssueResource matching the state of the issue.
129      * @param type the issue type
130      * @param priority the issue priority
131      * @param state the issue state
132      * @return an issue resource
133      */
134 	public IssueResource getState(String type, String priority, String state);
135 
136 	/**
137 	 * Returns a set of IssueResources representing user selectable states for defect issue types.
138 	 * @return
139 	 */
140 	public Set<? extends IssueResource> getDefaultDefectStates();
141 
142 	/**
143 	 * Returns a set of IssueResources representing user selectable states for all issue types. 
144 	 * @return
145 	 */
146 	public Set<? extends IssueResource> getDefaultStates();
147     	
148 	/**
149 	 * Return the default state for uninstall tracking.
150 	 */
151 	public IssueResource getUninstallState();
152 }
