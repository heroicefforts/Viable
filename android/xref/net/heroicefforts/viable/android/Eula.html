21  import java.io.BufferedReader;
22  import java.io.Closeable;
23  import java.io.IOException;
24  import java.io.InputStreamReader;
25  
26  import android.app.Activity;
27  import android.app.AlertDialog;
28  import android.content.DialogInterface;
29  import android.content.SharedPreferences;
30  import android.util.Log;

32  /**
33   * Displays an EULA ("End User License Agreement") that the user has to accept before
34   * using the application. Your application should call {@link Eula#show(android.app.Activity)}
35   * in the onCreate() method of the first activity. If the user accepts the EULA, it will never
36   * be shown again. If the user refuses, {@link android.app.Activity#finish()} is invoked
37   * on your activity.
38   */
39  class Eula {
40  	private static final String TAG = "Eula";
41  	
42      private static final String ASSET_EULA = "EULA";
43      private static final String PREFERENCE_EULA_ACCEPTED = "eula.accepted";
44      private static final String PREFERENCES_EULA = "eula";
45  
46      /**
47       * callback to let the activity know when the user has accepted the EULA.
48       */
49      static interface OnEulaAgreedTo {
50  
51          /**
52           * Called when the user has accepted the eula and the dialog closes.
53           */
54          void onEulaAgreedTo();
55      }
56  
57      /**
58       * Displays the EULA if necessary. This method should be called from the onCreate()
59       * method of your main Activity.
60       *
61       * @param activity The Activity to finish if the user rejects the EULA.
62       * @return Whether the user has agreed already.
63       */
64      static boolean show(final Activity activity) {
65          final SharedPreferences preferences = activity.getSharedPreferences(PREFERENCES_EULA,
66                  Activity.MODE_PRIVATE);
67          if (!preferences.getBoolean(PREFERENCE_EULA_ACCEPTED, false)) {
68              final AlertDialog.Builder builder = new AlertDialog.Builder(activity);
69              builder.setTitle(R.string.eula_title);
70              builder.setCancelable(true);
71              builder.setPositiveButton(R.string.eula_accept, new DialogInterface.OnClickListener() {
72                  public void onClick(DialogInterface dialog, int which) {
73                      accept(preferences);
74                      if (activity instanceof OnEulaAgreedTo) {
75                          ((OnEulaAgreedTo) activity).onEulaAgreedTo();
76                      }
77                  }
78              });
79              builder.setNegativeButton(R.string.eula_refuse, new DialogInterface.OnClickListener() {
80                  public void onClick(DialogInterface dialog, int which) {
81                      refuse(activity);
82                  }
83              });
84              builder.setOnCancelListener(new DialogInterface.OnCancelListener() {
85                  public void onCancel(DialogInterface dialog) {
86                      refuse(activity);
87                  }
88              });
89              builder.setMessage(readEula(activity));
90              builder.create().show();
91              return false;
92          }
93          return true;
94      }
95  
96      private static void accept(SharedPreferences preferences) {
97          preferences.edit().putBoolean(PREFERENCE_EULA_ACCEPTED, true).commit();
98      }
99  
100     private static void refuse(Activity activity) {
101         activity.finish();
102     }
103 
104     private static CharSequence readEula(Activity activity) {
105         BufferedReader in = null;
106         try {
107             in = new BufferedReader(new InputStreamReader(activity.getAssets().open(ASSET_EULA)));
108             String line;
109             StringBuilder buffer = new StringBuilder();
110             while ((line = in.readLine()) != null) buffer.append(line).append('\n');
111             return buffer;
112         } catch (IOException e) {
113             return "";
114         } finally {
115             closeStream(in);
116         }
117     }
118 
119     /**
120      * Closes the specified stream.
121      *
122      * @param stream The stream to close.
123      */
124     private static void closeStream(Closeable stream) {
125         if (stream != null) {
126             try {
127                 stream.close();
128             } catch (IOException e) {
129                 Log.e(TAG, "Error closing EULA input file stream.", e);
130             }
131         }
132     }
