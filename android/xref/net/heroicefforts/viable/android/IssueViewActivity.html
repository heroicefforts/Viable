168     /**
169      * Binds the issue data to the view
170      * @param issue the issue to display
171      * @throws ServiceException if there is an error connecting to the remote repository configuration.
172      */
173     private void presentIssue(Issue issue) 
174     	throws ServiceException
175 	{
176     	IssueResource issueState = factory.getRepository(issue.getAppName()).getState(issue.getType(), issue.getPriority(), issue.getState());
177 		TextView issueView = (TextView) findViewById(R.id.IssueIdTextView);
178 		issueView.setText(issue.getIssueId());
179 		issueView.setTextColor(Color.WHITE);
180 		((ImageView) findViewById(R.id.TypeImageView)).setImageDrawable(issueState.getIcon(this));
181 		
182 		TextView typeView = (TextView) findViewById(R.id.TypeTextView);
183 		typeView.setText(issueState.getTypeText(issue));
184 		
185 		TextView priorityView = (TextView) findViewById(R.id.PriorityTextView);
186 		priorityView.setText(issueState.getPriorityText(issue));
187 
188 		TextView stateView = (TextView) findViewById(R.id.StateTextView);
189 		stateView.setText(issueState.getStateText(issue));
190 		
191 		SpannableStringBuilder affectedVersions = new SpannableStringBuilder();
192 		affectedVersions.append(getString(R.string.affected_versions_label));
193 		String currVersion = factory.getApplicationVersion(issue.getAppName());
194 		if(Config.LOGD)
195 			Log.d(TAG, "Current version:  " + currVersion);
196 		for(String version : issue.getAffectedVersions())
197 		{
198 			if(version.equals(currVersion))
199 			{
200 				int idxStart = affectedVersions.length();
201 				affectedVersions.append(version);
202 				affectedVersions.setSpan(new ForegroundColorSpan(Color.RED), idxStart, affectedVersions.length(), Spanned.SPAN_EXCLUSIVE_EXCLUSIVE);
203 				affectedVersions.append(", ");
204 			}
205 			else
206 				affectedVersions.append(version).append(", ");
207 		}
208 		affectedVersions.delete(affectedVersions.length() - 2, affectedVersions.length() - 1);
209 		
210 		((TextView) findViewById(R.id.AffectedVersionsTextView)).setText(affectedVersions);
211 
212 		DateFormat fmt = DateFormat.getDateInstance();
213 		((TextView) findViewById(R.id.CreatedDateTextView)).setText(getString(R.string.created_date_label) + fmt.format(issue.getCreateDate()));
214 		((TextView) findViewById(R.id.ModifiedDateTextView)).setText(getString(R.string.modified_date_label) + fmt.format(issue.getModifiedDate()));
215 		((TextView) findViewById(R.id.SummaryTextView)).setText(issue.getSummary());
216 		((TextView) findViewById(R.id.DescriptonTextView)).setText(issue.getDescription());
217 		
218 		RemoteCommentCursor cursor = new RemoteCommentCursor(factory.getRepository(issue.getAppName()), issue.getIssueId());
219 		SimpleCursorAdapter adapter = new SimpleCursorAdapter(this, R.layout.comments_list_item, cursor,
220 		        new String[] { Comments.AUTHOR, Comments.BODY, Comments.CREATED_DATE }, 
221 		        new int[] { R.id.AuthorTextView, R.id.BodyTextView, R.id.CreateDateTextView });
222 		((ListView) findViewById(R.id.CommentsListView)).setAdapter(adapter);
223 		
224 		Cursor c = getContentResolver().query(Issues.CONTENT_URI, Issues.ISSUE_PROJECTION, Issues.ISSUE_ID + " = ?", new String[] { issue.getIssueId() }, null);
225 		Boolean voted = null;
226 		if(c.moveToFirst())
227 		{
228 			Issue localIssue = new IssueContentAdapter(c).toIssue();
229 			voted = localIssue.isVoted();
230 		}
231 		
232 		if(voted == null)
233 			voted = issue.isVoted();
234 		
235 		ImageView iv = (ImageView) findViewById(R.id.VotedImageView);
236 		if(voted)
237 			iv.setImageResource(android.R.drawable.star_big_on);
238 		else
239 		{
240 			iv.setImageResource(android.R.drawable.star_big_off);
241 			iv.setOnClickListener(starClicked);
242 		}
243 		
244 		TextView votesView = (TextView) findViewById(R.id.VotesTextView);
245 		votesView.setText(String.valueOf(issue.getVotes()));
246 		
247 
248 		((TextView) findViewById(R.id.VotesTextView)).setText(String.valueOf(issue.getVotes()));
249 
250 	}

261 	/**
262      * Results class for returning asynchronous result values.
263      */
264 	private class VoteHolder
265 	{
266 		public Issue issue;
267 		public Boolean success;
268 		public ServiceException exc;
269 	}
270     
271 	public class AsyncVoteTask extends AsyncTask<String, Void, VoteHolder>
272 	{
273 
274 		@Override
275 		protected VoteHolder doInBackground(String...args)
276 		{
277 			VoteHolder holder = new VoteHolder();
278 			try
279 			{
280 				String issueId = args[0];
281 				Repository rep = factory.getRepository(appName);
282 				holder.issue = rep.findById(issueId);
283 				holder.success = rep.vote(holder.issue);
284 			}
285 			catch (ServiceException e)
286 			{
287 				holder.exc = e;
288 			}
289 			
290 			return holder;
291 		}
292 
293 		@Override
294 		protected void onPostExecute(VoteHolder holder)
295 		{
296 			if(holder.exc == null)
297 			{
298 				String issueId = IssueViewActivity.this.getIssueId();
299 				if(holder.success && issueId.equals(holder.issue.getIssueId()))
300 				{
301 					updateIssue(holder.issue);
302 
303 					TextView votesView = ((TextView) findViewById(R.id.VotesTextView));
304 					votesView.setText(String.valueOf(holder.issue.getVotes()));
305 					ImageView iv = ((ImageView) findViewById(R.id.VotedImageView));
306 					iv.setImageResource(android.R.drawable.star_big_on);
307 					iv.setOnClickListener(null);
308 				}
309 			}
310 			else
311 				Error.handle(IssueViewActivity.this, holder.exc);
312 		}
313 
314 	}
