118     public void onStart()
119     {
120     	super.onStart();
121     	
122     	Intent intent;
123     	if(getParent() != null && (getParent() instanceof IssueTabsActivity))
124     		intent = getParent().getIntent();
125     	else
126     		intent = getIntent();
127 
128     	if(handled != intent)
129     	{
130     		try
131 			{
132 				BugReportIntent bri = new BugReportIntent(intent);
133 				if(bri.isValid())
134 					setDefectState(bri.getAppName(), bri.getStacktrace());

138 			catch (ServiceException e)

147 		throws ServiceException
148 	{

164 	/**
165 	 * Check if the defect has already been reported to the application developers.
166 	 * 
167 	 * @param ctx the defect
168 	 * @throws ServiceException if there's an error contacting the remote repository.
169 	 */
170 	private void duplicateIssueCheck(Issue ctx) 
171 		throws ServiceException
172 	{
173     	Repository rep = factory.getRepository(ctx.getAppName());
174     	if(rep == null)
175     		abort("Application '" + ctx.getAppName() + "' is not properly configured...see errors above.  Shutting down Viable.");
176 
177 		Issue issue = rep.exists(ctx);
178 		curIssue = issue;
179     	if(issue != null) //issue recorded remotely
180     	{
181     		Cursor cur = this.getContentResolver().query(Issues.CONTENT_URI, new String[] { Issues.APP_NAME, Issues.ISSUE_ID, Issues.SUMMARY }, Issues.APP_NAME + " = ? and " + Issues.ISSUE_ID + " = ?", new String[] { ctx.getAppName(), issue.getIssueId() }, Issues.DEFAULT_SORT_ORDER);
182     		if(cur.moveToFirst()) //issue encountered locally
183     			abort("The user has already encountered issue '" + issue.getIssueId() + "'.  Closing the activity.");
184     		else
185     			showSupplementDialog(issue);        			
186     	}    		
187 	}

235 	/**
236 	 * Dumps the issue details collected by the system to a string.  Allow the user to see what will be submitted.
237 	 * @param issue the defect
238 	 * @return a pretty printed string of issue details.
239 	 */
240     protected String formatDetails(Issue issue)
241     {
242     	final String EOL = System.getProperty("line.separator");
243     	StringBuilder buf = new StringBuilder();
244     	buf.append(getString(R.string.app_label)).append(issue.getAppName()).append(EOL);
245     	if(issue instanceof BugContext)
246     	{
247 	    	if(issue.getAffectedVersions().length > 0)
248 	    		buf.append(getString(R.string.version_name)).append(issue.getAffectedVersions()[0]).append(EOL);
249 	    	buf.append(getString(R.string.phone_model)).append(Build.MODEL).append(EOL);
250 	    	buf.append(getString(R.string.phone_device)).append(Build.DEVICE).append(EOL);
251 	    	buf.append(getString(R.string.phone_sdk)).append(Build.VERSION.SDK_INT).append(EOL);
252     	}
253     	buf.append(getString(R.string.error)).append(issue.getStacktrace());
254     	
255     	return buf.toString();
256     }

312 			{

314 				{
315 					detailsButton.setText(getText(R.string.show_details));
316 					detailsScroll.setVisibility(View.INVISIBLE);
317 				}
318 				else
319 				{

321 					detailsScroll.setVisibility(View.VISIBLE);

324 			}

335 	private void reportBug(String appName, String summary, String desc, IssueResource resource)
336 	{
337 		curIssue = new BugContext();
338 		resource.setState(curIssue);
339 		curIssue.setAppName(appName);
340 		curIssue.setSummary(summary);
341 		curIssue.setDescription(desc);
342 		curIssue.setStacktrace(stacktrace);
343 		curIssue.setAffectedVersions(new String[] { factory.getApplicationVersion(appName) });
344 		
345 		new ReportBugTask().execute(curIssue);  						
346 	}

348 	/**
349 	 * Asynchronous task for submitting the issue comment to the remote repository. 
350 	 */
351 	private class ReportCommentTask extends AsyncTask<Comment, Void, ServiceException>
352 	{
353 		@Override
354 		protected ServiceException doInBackground(Comment... params)
355 		{
356 			try
357 			{
358 				Comment comment = params[0];
359 				int responseCode = factory.getRepository(curIssue.getAppName()).postIssueComment(curIssue, comment);
360 				saveMyIssue(responseCode, curIssue);
361 				BugReporterActivity.this.comment = false;
362 				curIssue = null;
363 				return null;
364 			}
365 			catch (ServiceException e)
366 			{
367 				return e;
368 			}			
369 		}
370 		
371 		protected void onPostExecute(ServiceException e)
372 		{
373 			if(e == null)
374 				clearState();
375 			else
376 				Error.handle(BugReporterActivity.this, e);				
377 		}
378 		
379 	}

381 	/**
382 	 * Save the issue to the local content repository.
383 	 * @param responseCode the post response code.
384 	 * @param curIssue the issue to save
385 	 */
386 	private void saveMyIssue(int responseCode, Issue curIssue)
387 	{
388 		if(HttpStatus.SC_CREATED == responseCode || HttpStatus.SC_OK == responseCode)
389 		{			
390 			ContentValues values = new IssueContentAdapter(curIssue).toContentValues();
391 			BugReporterActivity.this.getContentResolver().insert(Issues.CONTENT_URI, values);
392 		}
393 	}

395 	/**
396 	 * Asynchronous task for submitting the issue to the remote repository. 
397 	 */
398 	private class ReportBugTask extends AsyncTask<Issue, Void, ServiceException> 
399 	{
400 		@Override
401 		protected ServiceException doInBackground(Issue... params)
402 		{
403 			try {
404 				Issue curIssue = params[0];
405 				int responseCode = factory.getRepository(curIssue.getAppName()).postIssue(curIssue);
406 				saveMyIssue(responseCode, curIssue);
407 				curIssue = null;
408 				
409 				return null;
410 			}
411 			catch(ServiceException e)
412 			{
413 				return e;
414 			}
415 		}
416 
417 		protected void onPostExecute(ServiceException e)
418 		{
419 			if(e == null)
420 				clearState();				
421 			else
422 				Error.handle(BugReporterActivity.this, e);				
423 		}
