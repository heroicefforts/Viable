106     /**
107      * Asynchronous task for asynchronously requesting application information and later displaying it.
108      * 
109      * @author jevans
110      *
111      */
112     private class LoadAppStatsTask extends AsyncTask<String, Void, ProjectDetail> 
113     {
114     	private String appName;
115     	private Exception exc;
116     	
117     	public void onPreExecute()
118     	{
119     		progressBar.setVisibility(View.VISIBLE);
120     	}
121     	
122     	protected ProjectDetail doInBackground(String...appNames)
123     	{
124     		this.appName = appNames[0];
125     		try {
126     			return factory.getRepository(appName).getApplicationStats();
127     		}
128 			catch (CreateException e)
129 			{
130 				this.exc = e;
131 				return null;
132 			}
133 			catch (ServiceException e)
134 			{
135 				this.exc = e;
136 				return null;
137 			}
138     		
139     	}
140     	
141     	protected void onPostExecute(final ProjectDetail pd)
142     	{
143     		if(exc == null)
144     			showProjectDetails(pd, appName);
145     		else
146     			Error.handle(AppStatsActivity.this, exc);
147     		
148 			progressBar.setVisibility(View.GONE);			
149 		}
150 		
151 	}

153     /**
154      * Set the project details to the view.
155      * 
156      * @param pd the project details
157      * @param appName the application name of the project supplied
158      */
159 	private void showProjectDetails(final ProjectDetail pd, String appName)
160 	{
161 		if(pd == null)
162 		{
163 			Log.e(TAG, "Error retrieving app stats for '" + appName + "'.");
164 			return;
165 		}
166 		
167 		if(!TextUtils.isEmpty(pd.getDescription()))
168 			((TextView) findViewById(R.id.DescriptionTextView)).setText(pd.getDescription());
169 		else
170 			((TextView) findViewById(R.id.DescriptionTextView)).setText("");
171 		
172 		if(!TextUtils.isEmpty(pd.getLead()))
173 			((TextView) findViewById(R.id.LeadTextView)).setText(getString(R.string.maintainer) + pd.getLead());
174 		else
175 			((TextView) findViewById(R.id.LeadTextView)).setText("");
176 		
177 		if(!TextUtils.isEmpty(pd.getUrl()))
178 		{
179 			SpannableStringBuilder urlSpan = new SpannableStringBuilder();
180 			urlSpan.append(getString(R.string.website));
181 			urlSpan.setSpan(new URLSpan(pd.getUrl()), 0, urlSpan.length(), Spanned.SPAN_EXCLUSIVE_EXCLUSIVE);
182 			((TextView) findViewById(R.id.URLTextView)).setText(urlSpan);
183 			((TextView) findViewById(R.id.URLTextView)).setOnClickListener(new OnClickListener() {
184 
185 				public void onClick(View v)
186 				{
187 					Intent intent = new Intent(Intent.ACTION_VIEW, Uri.parse(pd.getUrl()));		
188 					AppStatsActivity.this.startActivity(intent);
189 				}
190 				
191 			});
192 			Linkify.addLinks((TextView) findViewById(R.id.URLTextView), Linkify.ALL);
193 		}
194 		else
195 			((TextView) findViewById(R.id.URLTextView)).setText("");
196 
197 		setIssueCounts(R.id.BugsUnresTextView, R.id.BugsResTextView, pd.getUnfixedBugs(), pd.getFixedBugs());
198 		((TextView) findViewById(R.id.ImprsUnresTextView)).setText(String.valueOf(pd.getUnfixedImprovements()));
199 		((TextView) findViewById(R.id.ImprsResTextView)).setText(String.valueOf(pd.getFixedImprovements()));
200 		((TextView) findViewById(R.id.FeaturesUnresTextView)).setText(String.valueOf(pd.getUnfixedFeatures()));
201 		((TextView) findViewById(R.id.FeaturesResTextView)).setText(String.valueOf(pd.getFixedFeatures()));
202 		
203 		TableLayout vTable = (TableLayout) findViewById(R.id.VersionTableLayout);
204 		
205 		if(vTable.getChildCount() > 1)
206 			vTable.removeViews(1, vTable.getChildCount() - 1);
207 		DateFormat fmt = SimpleDateFormat.getDateInstance();
208 		for(VersionDetail version : pd.getVersions())
209 		{
210 			TableRow tr = new TableRow(AppStatsActivity.this);
211 			
212 			TextView name = new TextView(AppStatsActivity.this);
213 			SpannableStringBuilder nameStr = new SpannableStringBuilder();
214 			nameStr.append(version.getName());
215 			if(factory.getApplicationVersion(appName).equals(version.getName()))
216 				nameStr.setSpan(new ForegroundColorSpan(Color.RED), 0, nameStr.length(), Spanned.SPAN_EXCLUSIVE_EXCLUSIVE);
217 			name.setText(nameStr);			
218 			tr.addView(name);
219 			
220 			TextView date = new TextView(AppStatsActivity.this);
221 			//date.setLayoutParams(lp);
222 			if(version.getReleaseDate() != null)
223 				date.setText(fmt.format(version.getReleaseDate()));
224 			else
225 			{
226 				SpannableStringBuilder unreleased = new SpannableStringBuilder();
227 				unreleased.append(getString(R.string.unreleased));
228 				unreleased.setSpan(new ForegroundColorSpan(Color.BLUE), 0, unreleased.length(), Spanned.SPAN_EXCLUSIVE_EXCLUSIVE);				
229 				date.setText(unreleased);
230 			}
231 			tr.addView(date);
232 
233 			vTable.addView(tr);
234 			vTable.requestLayout();
235 		}		
236 	}

238 	/**
239 	 * Sets the vital bug counts for a row in the table view.
240 	 * 
241 	 * @param unresRes resource id for the unresolved text view 
242 	 * @param resRes resource id for the resolved text view
243 	 * @param unfixedCnt the unresolved issue count
244 	 * @param fixedCnt the fixed issue count
245 	 */
246 	private void setIssueCounts(int unresRes, int resRes, long unfixedCnt, long fixedCnt)
247 	{
248 		float pctFixed = ((float) fixedCnt) / unfixedCnt;
249 
250 		SpannableStringBuilder unfixedSpan = new SpannableStringBuilder();
251 		unfixedSpan.append(String.valueOf(unfixedCnt));
252 		if(pctFixed < .25)
253 			unfixedSpan.setSpan(new ForegroundColorSpan(Color.RED), 0, unfixedSpan.length(), Spanned.SPAN_EXCLUSIVE_EXCLUSIVE);
254 		else if(pctFixed < .5)
255 			unfixedSpan.setSpan(new ForegroundColorSpan(Color.YELLOW), 0, unfixedSpan.length(), Spanned.SPAN_EXCLUSIVE_EXCLUSIVE);		
256 		else if(pctFixed < .75)
257 			unfixedSpan.setSpan(new ForegroundColorSpan(Color.WHITE), 0, unfixedSpan.length(), Spanned.SPAN_EXCLUSIVE_EXCLUSIVE);
258 		
259 		((TextView) findViewById(unresRes)).setText(unfixedSpan);
260 		((TextView) findViewById(resRes)).setText(String.valueOf(fixedCnt));
261 	}
