45  /**
46   * This Activity class handles user uninstallation comments and submits them to the appropriate issue tracker belonging to the
47   * uninstalled application.  Comments are accumulated under one well-defined issue.
48   * 
49   * @author jevans
50   *
51   */
52  public class UninstallActivity extends Activity
53  {
54  	private static final String TAG = "BugReporterActivity";
55  	private static final String EOL = System.getProperty("line.separator");

69      @Override
70      public void onCreate(Bundle savedInstanceState) 
71      {
72          super.onCreate(savedInstanceState);
73          
74          UninstallIntent uninstall = new UninstallIntent(getIntent());
75          if(!uninstall.isValid())
76          {
77          	Log.e(TAG, "Uninstall activity did not contain sufficient intent data.  Was it invoked from the uninstall receiver?");
78          	finish();
79          }
80  
81          entry = uninstall.getAppEntry();
82          repository = new RepositoryFactory(this).instantiateRepository(entry);
83  		
84          try
85  		{
86  			initializeIssue();
87  		}
88  		catch (ServiceException e)
89  		{
90  			Log.e(TAG, "Error contacting repository.  Will not prompt for uninstall comments.");
91  			finish();
92  		}    		
93          
94          setContentView(R.layout.bug_report_activity);
95              	
96          ((Button) findViewById(R.id.ReportBugButton)).setOnClickListener(clickListener);
97          
98          appNameSpinner = (Spinner) findViewById(R.id.AppNameSpinner);    	
99      	typeSpinner = (Spinner) findViewById(R.id.IssueTypeSpinner);    	
100 		summaryText = (TextView) findViewById(R.id.BugSummaryEditText);
101 		descriptionText = (TextView) findViewById(R.id.BugDescriptionEditText);
102         detailsButton = (Button) findViewById(R.id.DetailsButton);
103         detailsScroll = (View) findViewById(R.id.BugDetailsScrollView);
104         
105 		List<String> appNames = new ArrayList<String>();
106 		appNames.add(entry.getAppName());
107     	appNameSpinner.setAdapter(new ArrayAdapter<String>(this, android.R.layout.simple_spinner_item, appNames));
108     	appNameSpinner.setEnabled(false);
109 
110     	((TextView) findViewById(R.id.IssueTypeTextView)).setVisibility(View.GONE);
111     	typeSpinner.setVisibility(View.GONE);
112 		detailsButton.setVisibility(View.GONE);
113 		detailsScroll.setVisibility(View.GONE);
114     }

124 	private void initializeIssue() 
125 		throws ServiceException
126 	{
127 		issue = new Issue();
128 		issue.setAppName(entry.getAppName());
129 		issue.setStacktrace("uninstall");
130 		issue.setAffectedVersions(new String[] { entry.getVersionName() });
131 		Issue existant = repository.exists(issue);
132     	if(existant != null) //issue recorded remotely
133     		issue = existant;    	
134 	}

139 	private void promptToComment()
140 	{
141 		if(Config.LOGD)
142 			Log.d(TAG, "Prompting user to comment on uninstall of '" + entry.getAppName() + "'.");
143 		final AlertDialog.Builder builder = new AlertDialog.Builder(this);
144 		builder.setTitle(R.string.uninstall_comment);
145 		builder.setCancelable(true);
146 		builder.setPositiveButton(R.string.yes, new DialogInterface.OnClickListener() {
147 		    public void onClick(DialogInterface dialog, int which) {
148 		    }
149 		});
150 		builder.setNegativeButton(R.string.no, new DialogInterface.OnClickListener() {
151 		    public void onClick(DialogInterface dialog, int which) {
152 		    	finish();
153 		    }
154 		});
155 		builder.setOnCancelListener(new DialogInterface.OnCancelListener() {
156 		    public void onCancel(DialogInterface dialog) {
157 		    	finish();
158 		    }
159 		});
160 		builder.setMessage(getString(R.string.uninstall_comment_prompt).replaceAll("\\{appName\\}", entry.getAppName()));
161 		builder.create().show();
162 	}

164 	private OnClickListener clickListener = new OnClickListener()
165 	{
166 
167 		public void onClick(View v)
168 		{
169 			if(v.getId() == R.id.ReportBugButton)
170 			{
171 				String summary = summaryText.getText().toString();
172 		        String desc = descriptionText.getText().toString();
173 
174 		        if(TextUtils.isEmpty(summary))
175 		        {
176 					summaryText.requestFocus();
177 					return;
178 		        }
179 
180 		        if(issue.getIssueId() != null)
181 		        	reportComment(summary, desc);
182 		        else
183 		        {
184 		        	issue.setSummary(getString(R.string.uninstall_reasons));
185 		        	issue.setDescription(summary + EOL + EOL + desc);
186 		        	repository.getUninstallState().setState(issue);
187 		        	reportBug(issue);
188 		        }			        
189 			}
190 		}
191 
192 	};

194 	private void reportComment(String summary, String desc)
195 	{			
196 		Comment comment = new Comment(summary + EOL + EOL + desc);
197 		new ReportCommentTask().execute(comment);
198 	}
199 
200 	private void reportBug(Issue issue)
201 	{		
202 		new ReportBugTask().execute(issue);  						
203 	}

205 	/**
206 	 * Asynchronous task for submitting the issue comment to the remote repository. 
207 	 */
208 	private class ReportCommentTask extends AsyncTask<Comment, Void, ServiceException>
209 	{
210 		@Override
211 		protected ServiceException doInBackground(Comment... params)
212 		{
213 			try
214 			{
215 				Comment comment = params[0];
216 				repository.postIssueComment(issue, comment);
217 				return null;
218 			}
219 			catch (ServiceException e)
220 			{
221 				return e;
222 			}			
223 		}
224 		
225 		protected void onPostExecute(ServiceException e)
226 		{
227 			if(e == null)
228 				finish();
229 			else
230 				Error.handle(UninstallActivity.this, e);				
231 		}
232 		
233 	}

235 	/**
236 	 * Asynchronous task for submitting the issue to the remote repository. 
237 	 */
238 	private class ReportBugTask extends AsyncTask<Issue, Void, ServiceException> 
239 	{
240 		@Override
241 		protected ServiceException doInBackground(Issue... params)
242 		{
243 			try {
244 				Issue curIssue = params[0];
245 				repository.postIssue(curIssue);				
246 				return null;
247 			}
248 			catch(ServiceException e)
249 			{
250 				return e;
251 			}
252 		}
253 
254 		protected void onPostExecute(ServiceException e)
255 		{
256 			if(e == null)
257 				finish();				
258 			else
259 				Error.handle(UninstallActivity.this, e);				
260 		}
261 
262 	}
