21  import android.app.Activity;
22  import android.app.AlertDialog;
23  import android.app.Dialog;
24  import android.content.BroadcastReceiver;
25  import android.content.Context;
26  import android.content.DialogInterface;
27  import android.content.Intent;
28  import android.content.IntentFilter;
29  import android.net.ConnectivityManager;
30  import android.net.NetworkInfo;
31  import android.util.Log;

33  /**
34   * Monitors network connectivity.  If connectivity is lost, then a modal dialog is raised prompting the user to 
35   * re-establish connectivity or close the application.  If connectivity is restored, then the dialog will close.  If the user
36   * chooses to end the application, then giveUp will be invoked on the parent activity.
37   * 
38   * This class supports delayed network dependent activity creation logic by invoking initOnNetworkAvailability.  
39   * 
40   * @author jevans
41   *
42   */
43  public class NetworkDependentDialog extends BroadcastReceiver
44  {
45  	private static final String TAG = "NetworkDependentDialog";
46  	
47  	private NetworkDependentActivity parent;
48  	protected AlertDialog dialog;
49  	private Activity activity;
50  	private boolean init;
51  
52  	/**
53  	 * Constructor.  Takes an Activity that implements NetworkDependentActivity.
54  	 * 
55  	 * @param parent the parent to which the modal dialog will bind.
56  	 */
57  	public NetworkDependentDialog(NetworkDependentActivity parent)
58  	{
59  		this.activity = verifyNonNullActivity(parent);
60  		this.parent = parent;
61  
62  		activity.registerReceiver(this, new IntentFilter(ConnectivityManager.CONNECTIVITY_ACTION));
63  		
64  		if(disconnected())
65  		{
66  			activity.setVisible(false);
67  			showNetworkDialog();
68  		}
69  		else
70  			networkEstablished();
71  	}
72  	
73  	/**
74  	 * Validate that the activity is not null and is an Activity.
75  	 * @param parent the dependent activity
76  	 * @return the casted activity.
77  	 */
78  	private Activity verifyNonNullActivity(NetworkDependentActivity parent)
79  	{
80  		if(parent == null)
81  			throw new IllegalArgumentException("The NetworkDependentActivity entity must not be null.");
82  		if(!(parent instanceof Activity))
83  			throw new IllegalArgumentException("Implementors of NetworkDependentActivity must extend Activity.");
84  		else
85  			return (Activity) parent;
86  	}
87  
88  	/**
89  	 * Deallocates resources.  This method should be invoked by the parent's finish method. 
90  	 */
91  	public void finish()
92  	{
93  		activity.unregisterReceiver(this);
94  	}
95  
96  	/**
97  	 * Cleans up the dialog resources and invokes the parent to giveUp.
98  	 */
99  	private void giveUp()
100 	{
101 		finish();
102 		parent.giveUp();		
103 	}
104 	
105 	/**
106 	 * Invoked by the system to notify of a network connectivity change broadcast.
107 	 * 
108 	 * @param context the system context
109 	 * @param intent the broadcast intention.
110 	 */
111 	@Override
112 	public void onReceive(Context context, Intent intent)
113 	{
114 		if (intent.getBooleanExtra(ConnectivityManager.EXTRA_NO_CONNECTIVITY, false))
115 		{
116 			activity.runOnUiThread(new Runnable() {
117 				public void run()
118 				{
119 					if(dialog == null)
120 						showNetworkDialog();
121 				}
122 			});
123 		}
124 		else
125 		{
126 			activity.runOnUiThread(new Runnable() {
127 				public void run()
128 				{
129 					if(dialog != null)
130 						dialog.getButton(Dialog.BUTTON_POSITIVE).performClick();
131 				}
132 			});				
133 		}
134 	}
135 
136 	/**
137 	 * Instantiates the dialog and displays the network unavailable dialog box.
138 	 */
139 	protected void showNetworkDialog()
140 	{
141         final AlertDialog.Builder builder = new AlertDialog.Builder(activity);
142         builder.setTitle(R.string.disconnected_title);
143         builder.setCancelable(true);
144         builder.setPositiveButton(R.string.retry, new DialogInterface.OnClickListener() {
145             public void onClick(DialogInterface dialog, int which) {
146             	if(!disconnected())
147             	{
148             		networkEstablished();
149             	}
150             	else
151             	{
152             		dialog.dismiss();
153             		showNetworkDialog();
154             	}
155             }
156 
157         });
158         builder.setNegativeButton(R.string.close, new DialogInterface.OnClickListener() {
159             public void onClick(DialogInterface dialog, int which) {
160                 giveUp();
161             }
162         });
163         builder.setOnCancelListener(new DialogInterface.OnCancelListener() {
164             public void onCancel(DialogInterface dialog) {
165             	if(disconnected())
166             		giveUp();
167             	else
168             	{
169             		networkEstablished();
170             	}
171             }
172         });
173         builder.setMessage(activity.getString(R.string.disconnected_msg));
174         dialog = builder.create();
175         dialog.show();		
176 	}
177 	
178 	/**
179 	 * Invokes the necessary logic when network availability exists or reappears.
180 	 */
181 	private void networkEstablished()
182 	{
183 		if(!init)
184 		{
185 			parent.initOnNetworkAvailability();
186 			init = true;
187 		}
188 		if(dialog != null)
189 		{
190 			try {
191 				dialog.dismiss();
192 			}
193 			catch(IllegalArgumentException iae)
194 			{
195 				Log.e(TAG, "Error dismissing dialog.", iae);
196 			}
197 			dialog = null;
198 		}
199 		activity.setVisible(true);
200 
201 	}
202 	
203 	/**
204 	 * @return true if no network connectivity is available.
205 	 */
206 	protected boolean disconnected()
207 	{
208 		ConnectivityManager connMgr = (ConnectivityManager) activity.getSystemService(Context.CONNECTIVITY_SERVICE);
209 		NetworkInfo net = connMgr.getActiveNetworkInfo();
210 		return net == null || !net.isAvailable();
211 	}
