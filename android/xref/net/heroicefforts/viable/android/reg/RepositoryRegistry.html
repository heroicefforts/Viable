36  /**
37   * This class handles the persistence of Viable client applications' configuration information.  This was required to fulfill
38   * the uninstall reporting requirement, as Android removes package data before the uninstall event is triggered.
39   * 
40   * @author jevans
41   *
42   */
43  public class RepositoryRegistry
44  {
45  	private static final String TAG = "RepositoryRegistry";
46  	
47  	private RepoDB registry;
48  	private Context ctx;
49  
50  
51  	/**
52  	 * Create an instance of the repository.
53  	 * 
54  	 * @param ctx a Viable context.
55  	 */
56  	public RepositoryRegistry(Context ctx)
57  	{
58  		this.ctx = ctx;
59  		this.registry = new RepoDB(ctx);
60  	}
61  	
62  	/**
63  	 * Save or update the registry entry for the specified uid.  If the supplied uid is not a
64  	 * Viable client application, then no action is taken.
65  	 * 
66  	 * @param uid the system assigned uid
67  	 */
68  	public void modify(int uid)
69  	{
70  		ArrayList<RegEntry> entries = getEntries(uid);
71  		for(RegEntry entry : entries)
72  			modify(entry);
73  	}
74  	
75  	/**
76  	 * Return the entry for the specified Viable client application's uid
77  	 * @param uid the system assigned uid
78  	 * @return the client app's entry, or null if the uid does not exist or represents an app
79  	 * that is not a Viable client application. 
80  	 */
81  	public RegEntry getEntryForUID(int uid)
82  	{
83  		return registry.getRegEntry(uid);
84  	}
85  
86  	/**
87  	 * Delete any entry related to the specified uid, if it exists.
88  	 * @param uid the system assigned uid
89  	 */
90  	public void deleteForUID(int uid)
91  	{
92  		registry.delete(uid);
93  	}
94  
95  	/**
96  	 * Retrieves an entry for the specified application label.
97  	 * @param appName the client application's label.
98  	 * @return the entry or null if no entry has been registered for the application. 
99  	 */
100 	public RegEntry getEntryForApp(String appName)
101 	{
102 		Integer uid = registry.getUIDByName(appName);
103 		if(uid != null)
104 			return registry.getRegEntry(uid);
105 		else
106 			return null;
107 	}

109 	private void modify(RegEntry entry)
110 		throws SQLException
111 	{
112 		if(isViableApp(entry))
113 		{
114 			registry.delete(entry.getUID());
115 			registry.save(entry);
116 		}
117 	}
118 		
119 	private boolean isViableApp(RegEntry entry)
120 	{
121 		return entry.getParams() != null && entry.getParams().getString("viable-provider") != null;
122 	}
123 	
124 	private ArrayList<RegEntry> getEntries(int uid)
125 	{
126 		ArrayList<RegEntry> apps = new ArrayList<RegEntry>();
127 		
128 		if(uid > -1)
129 		{
130 			PackageManager pkgMgr = ctx.getPackageManager();
131 			String[] pkgs = pkgMgr.getPackagesForUid(uid);
132 			if(pkgs != null)
133 			{
134 				for(String pkg : pkgs)
135 				{
136 					try
137 					{
138 						ApplicationInfo info = pkgMgr.getApplicationInfo(pkg, PackageManager.GET_META_DATA);
139 						String appName = pkgMgr.getApplicationLabel(info).toString();
140 						PackageInfo pkgInfo = pkgMgr.getPackageInfo(pkg, 0);
141 						String versionName = pkgInfo.versionName;
142 						apps.add(new RegEntry(uid, appName, versionName, info.metaData));
143 					}
144 					catch (NameNotFoundException e)
145 					{
146 						Log.e(TAG, "There was no application info defined for package '" + pkg + "'.");
147 					}
148 				}
149 			}
150 			else
151 				Log.e(TAG, "There was no package defined for newly installed application uid '" + uid + "'.");
152 		}
153 		else
154 			Log.e(TAG, "No uid was supplied with the installation extent!");
155 		
156 		return apps;
157 	}

160 	private class RepoDB extends SQLiteOpenHelper
161 	{
162 		private static final String DATABASE_NAME = "ViableRegistry";
163 		private static final int DATABASE_VERSION = 1;
164 		private static final String REGISTRY_TABLE_NAME = "viable_registry";
165 		private static final String SQL_INSERT = "insert into " + REGISTRY_TABLE_NAME + "(uid, key, value, created_date) values (?,?,?,?)";
166 
167 
168 		public RepoDB(Context ctx)
169 		{
170 			super(ctx, DATABASE_NAME, null, DATABASE_VERSION);
171 		}
172 		
173 		@Override
174 		public void onCreate(SQLiteDatabase db)
175 		{
176             db.execSQL("CREATE TABLE " + REGISTRY_TABLE_NAME + " ("
177                     + "_id INTEGER PRIMARY KEY,"
178                     + "uid INTEGER,"
179                     + "key TEXT NOT NULL ON CONFLICT ROLLBACK,"
180                     + "value TEXT NOT NULL ON CONFLICT ROLLBACK,"
181                     + "created_date INTEGER"
182                     + ");"); 
183 		}
184 		
185 		public void save(RegEntry entry)
186 			throws SQLException
187 		{
188 			Bundle bundle = entry.getParams();
189 
190 			if(bundle != null)
191 			{
192 				SQLiteDatabase db = getWritableDatabase();
193 				db.beginTransaction();
194 				try {
195 					long now = System.currentTimeMillis();
196 					long uid = (long) entry.getUID();
197 					
198 					db.execSQL(SQL_INSERT, new Object[] { uid, "app-name", entry.getAppName(), now });
199 					db.execSQL(SQL_INSERT, new Object[] { uid, "app-version-name", entry.getVersionName(), now });
200 					for(String key : bundle.keySet())
201 					{
202 						if(key.startsWith("viable"))
203 						{
204 							String value = bundle.getString(key);
205 							db.execSQL(SQL_INSERT, new Object[] { uid, key, value, now });
206 						}
207 					}
208 					db.setTransactionSuccessful();
209 				}
210 				finally {
211 					db.endTransaction();
212 
213 					if(db != null)
214 						db.close();
215 				}
216 			}
217 		}
218 		
219 		public void delete(int uid)
220 			throws SQLException
221 		{
222 			SQLiteDatabase db = getWritableDatabase();
223 			db.beginTransaction();
224 			try {
225 				db.execSQL("delete from " + REGISTRY_TABLE_NAME + " where uid = ?", 
226 					new Object[] { new Integer(uid).longValue() });
227 				db.setTransactionSuccessful();
228 			}
229 			finally {
230 				db.endTransaction();
231 
232 				if(db != null)
233 					db.close();
234 			}
235 		}

237 		public RegEntry getRegEntry(int uid)
238 		{
239 			SQLiteDatabase db = getReadableDatabase();
240 			Cursor c = null;
241 			String appName = null;
242 			String versionName = null;
243 			try {
244 				Bundle values = new Bundle();
245 				c = db.query(REGISTRY_TABLE_NAME, new String[] { "key", "value" }, "uid = ?", new String[] { String.valueOf(uid) }, null, null, null);
246 				while(c.moveToNext())
247 				{
248 					String key = c.getString(c.getColumnIndex("key"));
249 					String value = c.getString(c.getColumnIndex("value"));
250 					
251 					if("app-name".equals(key))
252 						appName = value;
253 					else if("app-version-name".equals(key))
254 						versionName = value;
255 					else
256 						values.putString(key, value);
257 				}
258 
259 				if(appName != null && versionName != null && values.keySet().size() > 0)
260 				{
261 					return new RegEntry(uid, appName, versionName, values);
262 				}
263 				else
264 					return null;
265 			}
266 			finally
267 			{
268 				if(c != null)
269 					c.close();
270 				if(db != null)
271 					db.close();
272 			}			
273 		}

275 		public Integer getUIDByName(String appName)
276 		{
277 			SQLiteDatabase db = getReadableDatabase();
278 			Cursor c = null;
279 			try {
280 				c = db.query(REGISTRY_TABLE_NAME, new String[] { "uid" }, "key = ? and value = ?", 
281 					new String[] { "app-name", appName }, null, null, null);
282 				if(c.moveToNext())
283 					return c.getInt(c.getColumnIndex("uid"));
284 				else
285 					return null;
286 			}
287 			finally
288 			{
289 				if(c != null)
290 					c.close();
291 				if(db != null)
292 					db.close();
293 			}			
294 		}
