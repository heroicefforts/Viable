30  /**
31   * This class retains the configuration information required to instantiate a repository instance for a particular Viable
32   * client application.  Data is acquired from the package manager and persisted to a database.
33   *  
34   * @author jevans
35   *
36   */
37  public class RegEntry implements Serializable, Externalizable
38  {
39  	private static final long serialVersionUID = 1L;
40  	
41  	private Bundle params;
42  	private String versionName;
43  	private String appName;
44  
45  	private int uid;
46  
47  
48  	/**
49  	 * Constructor created to meet Serialization demands.  It is not intended for developer use.
50  	 */
51  	public RegEntry()
52  	{
53  		//for serialization
54  	}
55  	
56  	/**
57  	 * Constructs an entry.
58  	 * @param uid the application uid
59  	 * @param appName the application's label
60  	 * @param versionName the application's version name.
61  	 * @param params the viable specific parameters.
62  	 */
63  	public RegEntry(int uid, String appName, String versionName, Bundle params)
64  	{
65  		this.uid = uid;
66  		this.appName = appName;
67  		this.versionName = versionName;
68  		this.params = params;
69  	}
70  
71  	/**
72  	 * Return viable specific configuration parameters.
73  	 * @return
74  	 */
75  	public Bundle getParams()
76  	{
77  		return params;
78  	}
79  
80  	/**
81  	 * Return the application's version name.
82  	 * @return
83  	 */
84  	public String getVersionName()
85  	{
86  		return versionName;
87  	}
88  
89  	/**
90  	 * Return the application's label.
91  	 * @return
92  	 */
93  	public String getAppName()
94  	{
95  		return appName;
96  	}
97  
98  
99  	@Override
100 	public int hashCode()
101 	{
102 		final int prime = 31;
103 		int result = 1;
104 		result = prime * result + ((appName == null) ? 0 : appName.hashCode());
105 		return result;
106 	}

109 	@Override
110 	public boolean equals(Object obj)
111 	{
112 		if (this == obj)
113 			return true;
114 		if (obj == null)
115 			return false;
116 		if (getClass() != obj.getClass())
117 			return false;
118 		RegEntry other = (RegEntry) obj;
119 		if (appName == null)
120 		{
121 			if (other.appName != null)
122 				return false;
123 		}
124 		else if (!appName.equals(other.appName))
125 			return false;
126 		return true;
127 	}

129 	/**

131 	 * @return
132 	 */
133 	public int getUID()
134 	{
135 		return uid;
136 	}

139 	@SuppressWarnings("unchecked")
140 	public void readExternal(ObjectInput in) throws IOException, ClassNotFoundException
141 	{
142 		this.uid = in.readInt();
143 		this.appName = (String) in.readObject();
144 		this.versionName = (String) in.readObject();
145 		HashMap<String, String> map = (HashMap<String, String>) in.readObject();
146 		this.params = new Bundle(map.size());
147 		for(String key : map.keySet())
148 			params.putString(key, map.get(key));
149 	}
150 
151 
152 	public void writeExternal(ObjectOutput out) throws IOException
153 	{
154 		out.writeInt(uid);
155 		out.writeObject(appName);
156 		out.writeObject(versionName);
157 		HashMap<String, String> map = new HashMap<String, String>(params.size());
158 		for(String key : params.keySet())
159 			map.put(key, params.getString(key));
160 		out.writeObject(map);
161 	}
