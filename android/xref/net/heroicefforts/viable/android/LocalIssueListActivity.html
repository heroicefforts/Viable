67  	protected List<String> getAppNames()
68  	{
69  		List<String> apps = new ArrayList<String>();
70      	apps.add(getString(R.string.all));
71      	Cursor appNameCursor = managedQuery(Issues.APP_CONTENT_URI, new String[] { Issues._ID, Issues.APP_NAME }, null, null, null);
72      	while(appNameCursor.moveToNext())
73      		apps.add(appNameCursor.getString(1));
74  
75  		return apps;
76  	}

78  	protected ArrayList<String> getVersionList(int position, String appName)
79  	{
80  		TreeSet<String> versions = new TreeSet<String>();
81  		Cursor cursor;
82  		if(position > 0)
83  			cursor = managedQuery(Issues.VERSION_CONTENT_URI, new String[] { Issues.APP_VERSION }, Issues.APP_NAME + " = ?", new String[] { appName }, Issues.DEFAULT_SORT_ORDER);
84  		else
85  			cursor = managedQuery(Issues.VERSION_CONTENT_URI, new String[] { Issues.APP_VERSION }, null, null, Issues.DEFAULT_SORT_ORDER);
86  		while(cursor.moveToNext())
87  		{
88  			String v = cursor.getString(cursor.getColumnIndex(Issues.APP_VERSION));
89  			v = v.substring(1, v.length() - 1);
90  			String[] vArr = v.split("[ ]*,[ ]*");
91  			for(int i = 0; i < vArr.length; i++)
92  				versions.add(vArr[i]);
93  		}
94  		ArrayList<String> retVal = new ArrayList<String>(versions);
95  		retVal.add(0, getString(R.string.all));
96  		return retVal;
97  
98  	}

100 	protected Cursor getIssueCursor(String appName, String version)
101 	{
102 	    final String[] PROJECTION = new String[] {
103             Issues._ID,
104             Issues.APP_NAME,
105             Issues.ISSUE_ID,
106             Issues.ISSUE_TYPE,
107             Issues.ISSUE_PRIORITY,
108             Issues.ISSUE_STATE,
109             Issues.VOTED,
110             Issues.VOTES,
111             Issues.SUMMARY
112 	    };
113 				
114 		Cursor cursor;
115 		if(appName == null && version == null)
116 			cursor = managedQuery(getIntent().getData(), PROJECTION, null, null, Issues.DEFAULT_SORT_ORDER);
117 		else if(version == null)
118 			cursor = managedQuery(getIntent().getData(), PROJECTION, Issues.APP_NAME + " = ?", new String[] { appName },
119 	                Issues.DEFAULT_SORT_ORDER);
120 		else if(appName == null)
121 			cursor = managedQuery(getIntent().getData(), PROJECTION, Issues.APP_VERSION + " like ?", new String[] { "%" + version + "%" },
122 	                Issues.DEFAULT_SORT_ORDER);			
123 		else
124 			cursor = managedQuery(getIntent().getData(), PROJECTION, Issues.APP_NAME + " = ? and " + Issues.APP_VERSION + " like ?", new String[] { appName, "%" + version + "%" },
125 	                Issues.DEFAULT_SORT_ORDER);
126 			
127 		return cursor;
128 	}

130     protected OnItemClickListener getIssueListClickListener()
131     {
132     	return new OnItemClickListener() 
133         {    	
134     	    public void onItemClick(AdapterView<?> l, View v, int position, long id)
135     		{
136     	        Uri uri = ContentUris.withAppendedId(getIntent().getData(), id);
137     	        
138     	        String action = getIntent().getAction();
139     	        if (Intent.ACTION_PICK.equals(action) || Intent.ACTION_GET_CONTENT.equals(action))
140     	            setResult(RESULT_OK, new Intent().setData(uri));
141     	        else
142     	            startActivity(new Intent(Intent.ACTION_EDIT, uri));
143     		}
144         };
145     }

147     @Override
148     public void onCreateContextMenu(ContextMenu menu, View view, ContextMenuInfo menuInfo) {
149         AdapterView.AdapterContextMenuInfo info;
150         try {
151              info = (AdapterView.AdapterContextMenuInfo) menuInfo;
152         } catch (ClassCastException e) {
153             Log.e(TAG, "bad menuInfo", e);
154             return;
155         }
156 
157         Cursor cursor = (Cursor) listView.getAdapter().getItem(info.position);
158         if (cursor == null) {
159             // For some reason the requested item isn't available, do nothing
160             return;
161         }
162 
163         // Setup the menu header
164         menu.setHeaderTitle(cursor.getString(cursor.getColumnIndex(Issues.ISSUE_ID)));
165 
166         // Add a menu item to delete the issue
167         menu.add(0, MENU_ITEM_DELETE, 0, R.string.menu_delete);
168     }
169         
170     @Override
171     public boolean onContextItemSelected(MenuItem item) {
172         AdapterView.AdapterContextMenuInfo info;
173         try {
174              info = (AdapterView.AdapterContextMenuInfo) item.getMenuInfo();
175         } catch (ClassCastException e) {
176             Log.e(TAG, "bad menuInfo", e);
177             return false;
178         }
179 
180         switch (item.getItemId()) {
181             case MENU_ITEM_DELETE: {
182                 Uri issueUri = ContentUris.withAppendedId(getIntent().getData(), info.id);
183                 getContentResolver().delete(issueUri, null, null);
184                 return true;
185             }
186         }
187         return false;
188     }
