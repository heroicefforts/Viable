117     @Override
118     public Cursor query(Uri uri, String[] projection, String selection, String[] selectionArgs,
119             String sortOrder) {
120         SQLiteQueryBuilder qb = new SQLiteQueryBuilder();
121 
122         switch (sUriMatcher.match(uri)) 
123         {
124 	        case ISSUES:
125 	            qb.setTables(ISSUES_TABLE_NAME);
126 	            qb.setProjectionMap(issuesProjectionMap);
127 	            break;
128 	
129 	        case ISSUES_ID:
130 	            qb.setTables(ISSUES_TABLE_NAME);
131 	            qb.setProjectionMap(issuesProjectionMap);
132 	            qb.appendWhere(Issues._ID + "=" + uri.getPathSegments().get(1));
133 	            break;
134 	
135 	        case APPS:
136 	        	qb.setTables(REG_APPS_TABLE_NAME);
137 	        	qb.setProjectionMap(appsProjectionMap);
138 	        	break;
139 	        
140 	        case VERSIONS:
141 	        	qb.setTables(ISSUES_TABLE_NAME);
142 	        	qb.setProjectionMap(versionsProjectionMap);
143 	        	qb.setDistinct(true);
144 	        	break;
145 	        	
146 	        default:
147 	            throw new IllegalArgumentException(MSG_UNKNOWN_URI_ERROR + uri);
148         }
149 
150         // If no sort order is specified use the default
151         String orderBy;
152         if (TextUtils.isEmpty(sortOrder)) {
153             orderBy = Issues.DEFAULT_SORT_ORDER;
154         } else {
155             orderBy = sortOrder;
156         }
157 
158         // Get the database and run the query
159         SQLiteDatabase db = openHelper.getReadableDatabase();
160         Cursor c = qb.query(db, projection, selection, selectionArgs, null, null, orderBy);
161 
162         // Tell the cursor what uri to watch, so it knows when its source data changes
163         c.setNotificationUri(getContext().getContentResolver(), uri);
164         return c;
165     }

184     @Override
185     public Uri insert(Uri uri, ContentValues initialValues) {
186         // Validate the requested uri
187     	int type = sUriMatcher.match(uri);
188         if (type != ISSUES && type != APPS) {
189             throw new IllegalArgumentException(MSG_UNKNOWN_URI_ERROR + uri);
190         }
191         
192         ContentValues values;
193         if (initialValues != null) {
194             values = new ContentValues(initialValues);
195         } else {
196             values = new ContentValues();
197         }
198 
199         Long now = Long.valueOf(System.currentTimeMillis());
200 
201         // Make sure that the fields are all set
202         if (values.containsKey(Issues.CREATED_DATE) == false) {
203             values.put(Issues.CREATED_DATE, now);
204         }
205 
206         if (values.containsKey(Issues.MODIFIED_DATE) == false) {
207             values.put(Issues.MODIFIED_DATE, now);
208         }
209 
210         SQLiteDatabase db = openHelper.getWritableDatabase();
211         long rowId;
212         if(type == ISSUES)
213         {
214         	Cursor apps = null;
215         	try {
216         		apps = db.query(REG_APPS_TABLE_NAME, new String[] { "_ID" }, Issues.APP_NAME + " = ?", new String[] { (String) values.get(Issues.APP_NAME) }, null, null, null);
217             	if(!apps.moveToFirst())
218             	{
219             		ContentValues appValues = new ContentValues();
220             		appValues.put(Issues.APP_NAME, values.getAsString(Issues.APP_NAME));
221                     values.put(Issues.CREATED_DATE, now);
222                     values.put(Issues.MODIFIED_DATE, now);
223             		long appRowId = db.insert(REG_APPS_TABLE_NAME, Issues.APP_NAME, appValues);
224             		if(appRowId > 0) 
225             		{
226                     	Uri newUri = ContentUris.withAppendedId(Issues.APP_CONTENT_URI, appRowId);            
227                         getContext().getContentResolver().notifyChange(newUri, null);            			
228             		}
229             	}
230         	}
231         	finally
232         	{
233         		if(apps != null)
234         			apps.close();
235         	}
236         	
237         	rowId = db.insert(ISSUES_TABLE_NAME, Issues.ISSUE_ID, values);
238         }
239         else
240         	rowId = db.insert(REG_APPS_TABLE_NAME, Issues.APP_NAME, values);
241         if (rowId > 0) {
242             Uri newUri;
243             if(type == ISSUES)
244             	newUri = ContentUris.withAppendedId(Issues.CONTENT_URI, rowId);
245             else
246             	newUri = ContentUris.withAppendedId(Issues.APP_CONTENT_URI, rowId);            
247             getContext().getContentResolver().notifyChange(newUri, null);
248             return newUri;
249         }
250 
251         throw new SQLException("Failed to insert row into " + uri);
252     }

254     @Override
255     public int delete(Uri uri, String where, String[] whereArgs) {
256         SQLiteDatabase db = openHelper.getWritableDatabase();
257         int count;
258         switch (sUriMatcher.match(uri)) {
259         case ISSUES:
260             count = db.delete(ISSUES_TABLE_NAME, where, whereArgs);
261             break;
262 
263         case ISSUES_ID:
264             String noteId = uri.getPathSegments().get(1);
265             count = db.delete(ISSUES_TABLE_NAME, Issues._ID + "=" + noteId
266                     + (!TextUtils.isEmpty(where) ? " AND (" + where + ')' : ""), whereArgs);
267             break;
268 
269         case APPS:
270         	count = db.delete(REG_APPS_TABLE_NAME, where, whereArgs);
271         	break;
272             
273         default:
274             throw new IllegalArgumentException(MSG_UNKNOWN_URI_ERROR + uri);
275         }
276 
277         getContext().getContentResolver().notifyChange(uri, null);
278         return count;
279     }

281     @Override
282     public int update(Uri uri, ContentValues values, String where, String[] whereArgs) {
283         SQLiteDatabase db = openHelper.getWritableDatabase();
284         int count;
285         switch (sUriMatcher.match(uri)) {
286         case ISSUES:
287             count = db.update(ISSUES_TABLE_NAME, values, where, whereArgs);
288             if(count == 0)
289             	db.insert(ISSUES_TABLE_NAME, null, values);
290             break;
291 
292         case ISSUES_ID:
293             String issueId = uri.getPathSegments().get(1);
294             count = db.update(ISSUES_TABLE_NAME, values, Issues._ID + "='" + issueId + "'"
295                     + (!TextUtils.isEmpty(where) ? " AND (" + where + ")" : ""), whereArgs);
296             break;
297 
298         case APPS:
299         	count = db.update(REG_APPS_TABLE_NAME, values, where, whereArgs);
300         	break;
301             
302         default:
303             throw new IllegalArgumentException(MSG_UNKNOWN_URI_ERROR + uri);
304         }
305 
306         getContext().getContentResolver().notifyChange(uri, null);
307         return count;
308     }
