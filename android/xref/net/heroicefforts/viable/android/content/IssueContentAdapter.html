19  package net.heroicefforts.viable.android.content;
20  
21  import java.util.Arrays;
22  import java.util.Date;
23  
24  import net.heroicefforts.viable.android.dao.Issue;
25  import android.content.ContentValues;
26  import android.database.Cursor;
27  
28  /**
29   * This class converts issues between object type and cursor type.
30   * 
31   * @author jevans
32   *
33   */
34  public class IssueContentAdapter
35  {
36  	private Issue issue;
37  
38  	
39  	public IssueContentAdapter(Issue issue)
40  	{
41  		this.issue = issue;
42  	}
43  	
44  	public IssueContentAdapter(Cursor cursor)
45  	{
46  		issue = new Issue();
47  		issue.setIssueId(cursor.getString(cursor.getColumnIndex(Issues.ISSUE_ID)));
48  		issue.setType(cursor.getString(cursor.getColumnIndex(Issues.ISSUE_TYPE)));
49  		issue.setPriority(cursor.getString(cursor.getColumnIndex(Issues.ISSUE_PRIORITY)));
50  		issue.setState(cursor.getString(cursor.getColumnIndex(Issues.ISSUE_STATE)));
51  		issue.setAppName(cursor.getString(cursor.getColumnIndex(Issues.APP_NAME)));
52  		issue.setSummary(cursor.getString(cursor.getColumnIndex(Issues.SUMMARY)));
53  		issue.setDescription(cursor.getString(cursor.getColumnIndex(Issues.DESCRIPTION)));
54  		String versionString = cursor.getString(cursor.getColumnIndex(Issues.APP_VERSION));
55  		versionString = versionString.substring(1, versionString.length() - 1);		
56  		issue.setAffectedVersions(versionString.split("[ ]*,[ ]*"));
57  		issue.setHash(cursor.getString(cursor.getColumnIndex(Issues.HASH)));
58  		issue.setStacktrace(cursor.getString(cursor.getColumnIndex(Issues.STACKTRACE)));
59  		issue.setVoted(cursor.getInt(cursor.getColumnIndex(Issues.VOTED)) > 0);
60  		issue.setVotes(cursor.getLong(cursor.getColumnIndex(Issues.VOTES)));
61  		issue.setCreateDate(new Date(cursor.getLong(cursor.getColumnIndex(Issues.CREATED_DATE))));
62  		issue.setModifiedDate(new Date(cursor.getLong(cursor.getColumnIndex(Issues.MODIFIED_DATE))));		
63  	}
64  	
65  	/**
66  	 * Convert the issue state to content persistable values.
67  	 * @return content values for the adapter's issue state.
68  	 */
69  	public ContentValues toContentValues()
70  	{
71  		ContentValues values = new ContentValues();
72  		values.put(Issues.ISSUE_ID, issue.getIssueId());
73  		values.put(Issues.APP_NAME, issue.getAppName());
74  		values.put(Issues.ISSUE_TYPE, issue.getType());
75  		values.put(Issues.ISSUE_PRIORITY, issue.getPriority());
76  		values.put(Issues.ISSUE_STATE, issue.getState());
77  		values.put(Issues.STACKTRACE, issue.getStacktrace());
78  		values.put(Issues.SUMMARY, issue.getSummary());
79  		values.put(Issues.DESCRIPTION, issue.getDescription());
80  		values.put(Issues.HASH, issue.getHash());
81  		values.put(Issues.VOTED, issue.isVoted() ? 1 : 0);
82  		values.put(Issues.VOTES, issue.getVotes());
83  		values.put(Issues.CREATED_DATE, issue.getCreateDate().getTime());
84  		values.put(Issues.MODIFIED_DATE, issue.getModifiedDate().getTime());
85  		values.put(Issues.APP_VERSION, Arrays.asList(issue.getAffectedVersions()).toString());
86  		
87  		return values;
88  	}
89  	
90  	/**
91  	 * Convert the issue state to an issue
92  	 * @return a populated issue
93  	 */
94  	public Issue toIssue()
95  	{
96  		return issue;		
97  	}
98  }
