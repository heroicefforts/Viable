19  package net.heroicefforts.viable.android;
20  
21  import android.app.TabActivity;
22  import android.content.Intent;
23  import android.os.Bundle;
24  import android.util.Log;
25  import android.widget.TabHost;
26  
27  /**
28   * The main activity for Viable.  It is invoked by the launcher and {@link CrashReceiver}.  It handles the instantiation 
29   * of all the subactivities.
30   *  
31   * @author jevans
32   *
33   */
34  public class IssueTabsActivity extends TabActivity implements NetworkDependentActivity
35  {
36  	public static final String TAB_APP_STATS = "apps";
37  	public static final String TAB_ALL_ISSUES = "all_issues";
38  	public static final String TAB_MY_ISSUES = "my_issues";
39  	public static final String TAB_REPORT_ISSUE = "report_issue";
40  
41  	/**
42  	 * May be included with an intent to set the default tab view using one of the TAB_* constants.
43  	 */
44  	public static final String EXTRA_DEFAULT_TAB = "default_tab";
45  
46  	private NetworkDependentDialog dialog;
47  
48  	@Override
49  	public void onCreate(Bundle savedStateInstance)
50  	{
51  		super.onCreate(savedStateInstance);
52  
53  		dialog = new NetworkDependentDialog(this);
54          Eula.show(this);
55  	}
56  	
57  	@Override
58  	public void finish()
59  	{
60  		super.finish();
61  		dialog.finish();
62  	}
63  
64  	/**
65  	 * Set the default tab using the override EXTRA from the intent, if necessary.
66  	 * @param intent
67  	 */
68  	private void setDefaultTab(Intent intent)
69  	{
70  		String defaultTab = intent.getStringExtra(EXTRA_DEFAULT_TAB);
71  		if(Config.LOGD) 
72  		{		
73  			Log.d("IssueTabsActivity", "Received intent:  " + intent.toString());
74  			Log.d("IssueTabsActivity", "Setting tab to '" + defaultTab + "'.");
75  		}
76  		if (defaultTab == null)
77  			defaultTab = TAB_APP_STATS;
78  		getTabHost().setCurrentTabByTag(defaultTab);
79  	}
80  
81  	/**
82  	 * Selects the "My Issues" tab.
83  	 */
84  	public void selectMyIssuesTab()
85  	{
86  		getTabHost().setCurrentTabByTag(TAB_MY_ISSUES);
87  	}
88  
89  	/**
90  	 * Delays instantiation of the dependent tab activities until networking is available.  This eliminates network errors from
91  	 * bubbling up to the UI unnecessarily. 
92  	 */
93  	public void initOnNetworkAvailability()
94  	{
95  		TabHost mTabHost = getTabHost();
96  		mTabHost.addTab(mTabHost.newTabSpec(TAB_APP_STATS).setIndicator(getString(R.string.apps), getResources().getDrawable(R.drawable.apps)).setContent(
97  				new Intent(this, AppStatsActivity.class)));
98  		mTabHost.addTab(mTabHost.newTabSpec(TAB_MY_ISSUES).setIndicator(getString(R.string.my_issues), getResources().getDrawable(R.drawable.my_issues))
99  				.setContent(new Intent(this, LocalIssueListActivity.class)));
100 		mTabHost.addTab(mTabHost.newTabSpec(TAB_ALL_ISSUES).setIndicator(getString(R.string.all_issues), getResources().getDrawable(R.drawable.all_issues))
101 				.setContent(new Intent(this, RemoteIssueListActivity.class)));
102 		mTabHost.addTab(mTabHost.newTabSpec(TAB_REPORT_ISSUE).setIndicator(getString(R.string.report_issue), getResources().getDrawable(R.drawable.pen))
103 				.setContent(new Intent(this, BugReporterActivity.class)));
104 		setDefaultTab(getIntent());		
105 	}
106 	
107 	/**
108 	 * Closes this activity.
109 	 */
110 	public void giveUp()
111 	{
112 		finish();
113 	}	
114 
115 }
