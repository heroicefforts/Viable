54  	/**
55  	 * Instantiate the project state based upon the JIRA JSON format.
56  	 * 
57  	 * @param obj JIRA JSON project object 
58  	 * @throws JSONException if there's an error parsing the JSON.
59  	 */
60  	public ProjectDetail(JSONObject obj)
61  		throws JSONException
62  	{
63  		this.name = obj.getString("name");
64  		this.description = obj.getString("description");
65  		this.lead = obj.getString("lead");
66  		this.url = obj.getString("url");
67  		this.unfixedBugs = obj.getLong("unfixedBugs");
68  		this.fixedBugs = obj.getLong("fixedBugs");
69  		this.unfixedImprovements = obj.getLong("unfixedImprovements");
70  		this.fixedImprovements = obj.getLong("fixedImprovements");
71  		this.unfixedFeatures = obj.getLong("unfixedFeatures");
72  		this.fixedFeatures = obj.getLong("fixedFeatures");
73  		
74  		if(obj.has("versions"))
75  		{
76  			JSONArray vers = obj.getJSONArray("versions");
77  			for(int i = 0; i < vers.length(); i++)
78  				versions.add(new VersionDetail(vers.getJSONObject(i)));
79  			Collections.sort(versions);
80  		}
81  	}
